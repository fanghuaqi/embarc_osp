104 static void perf_init(unsigned int id)
105 {
106  if (timer_start(id, TIMER_CTRL_NH, 0xFFFFFFFF) < 0) {
107  EMBARC_PRINTF("perf timer init failed\r\n");
108  while(1);
109  }
110  perf_id = id;
111 }

114 static void perf_start(void)
115 {
116  if (timer_current(perf_id, (void *)(&start)) < 0) {
117  start = 0;
118  }
119 }

122 static unsigned int perf_end(void)
123 {
124  unsigned int end = 0;
125 
126  if (timer_current(perf_id, (void *)(&end)) < 0) {
127  return 0;
128  }
129 
130  if (start < end) {
131  return (end - start);
132  } else {
133  return (0xFFFFFFFF - start + end);
134  }
135 }
