104 static void task_wifi(void *par)
105 {
106  WNIC_AUTH_KEY auth_key;
107  int flag=0;
108 
109  pmwifi_wnic = wnic_get_dev(BOARD_PMWIFI_0_ID);
110 
111 #if WF_HOTSPOT_IS_OPEN
112  auth_key.key = NULL;
113  auth_key.key_len = 0;
114  auth_key.key_idx = 0;
115 #else
116  auth_key.key = (const uint8_t *)WF_HOTSPOT_PASSWD;
117  auth_key.key_len = sizeof(WF_HOTSPOT_PASSWD);
118  auth_key.key_idx = 0;
119 #endif
120  lwip_pmwifi_init();
121  EMBARC_PRINTF("\r\nNow trying to connect to WIFI hotspot, please wait about 30s!\r\n");
122 
123  while (1) {
124  pmwifi_wnic->period_process(par);
125 #if WF_HOTSPOT_IS_OPEN
126  pmwifi_wnic->wnic_connect(AUTH_SECURITY_OPEN, (const uint8_t *)WF_HOTSPOT_NAME, &auth_key);
127 #else
128  pmwifi_wnic->wnic_connect(AUTH_SECURITY_WPA_AUTO_WITH_PASS_PHRASE, (const uint8_t *)WF_HOTSPOT_NAME, &auth_key);
129 #endif
130  if ((flag == 0) && lwip_pmwifi_isup()) {
131  flag = 1;
132  EMBARC_PRINTF("WiFi connected \r\n");
133 #ifndef MID_NTSHELL /* resume main task when ntshell task is not defined */
134  if (task_handle_main) vTaskResume(task_handle_main);
135 #else
136  EMBARC_PRINTF("Please run NT-Shell command(main) to start your application.\r\n");
137  EMBARC_PRINTF("main command may required some arguments, please refer to example's document.\r\n");
138 #endif
139  /* consider to generate a event to notify network is ready */
140  }
141  vTaskDelay(TASK_WIFI_PERIOD);
142  }
143 }
144 #endif

146 static void task_main(void *par)
147 {
148  int ercd;
149 #if defined(OS_FREERTOS) && defined(MID_LWIP) && !defined(MID_NTSHELL)
150  EMBARC_PRINTF("Enter to main function....\r\n");
151  EMBARC_PRINTF("Wait until WiFi connected...\r\n");
152  vTaskSuspend(NULL);
153 #endif
154 
155  if ((par == NULL) || (((int)par) & 0x3)) {
156  /* null or aligned not to 4 bytes */
157  ercd = _arc_goto_main(0, NULL);
158  } else {
159  MAIN_ARGS *main_arg = (MAIN_ARGS *)par;
160  ercd = _arc_goto_main(main_arg->argc, main_arg->argv);
161  }
162 
163 #if defined(OS_FREERTOS)
164  EMBARC_PRINTF("Exit from main function, error code:%d....\r\n", ercd);
165  while (1) {
166  vTaskSuspend(NULL);
167  }
168 #else
169  while (1);
170 #endif
171 }

173 void board_main(void)
174 {
175 /* board level hardware init */
176  board_init();
177 /* board level middlware init */

180 #ifdef MID_COMMON
181  xprintf_setup();

184 #ifdef MID_FATFS
185  if(f_mount(&sd_card_fs, "", 0) != FR_OK) {
186  EMBARC_PRINTF("FatFS failed to initialize!\r\n");
187  } else {
188  EMBARC_PRINTF("FatFS initialized successfully!\r\n");
189  }
190 #endif
191 
192 #ifdef ENABLE_OS
193  os_hal_exc_init();
194 #endif
195 
196 /* NTSHELL related initialization */
197 /* For OS situation, ntshell task will be created; for baremetal ntshell_task will be executed */
198 #ifdef MID_NTSHELL
199  NTSHELL_IO *nt_io;
200  nt_io = get_ntshell_io(BOARD_ONBOARD_NTSHELL_ID);
201 
202 #ifdef OS_FREERTOS
203  xTaskCreate((TaskFunction_t)ntshell_task, "ntshell-console", TASK_STACK_SIZE_NTSHELL,
204  (void *)nt_io, TASK_PRI_NTSHELL, &task_handle_ntshell);
205 #else
206  cpu_unlock(); /* unlock cpu to let interrupt work */
208  ntshell_task((void *)nt_io);
209 #endif
210 
211 #else /* No ntshell */
212 #ifdef OS_FREERTOS
213  xTaskCreate((TaskFunction_t)task_main, "main", TASK_STACK_SIZE_MAIN,
214  (void *)(&s_main_args), TASK_PRI_MAIN, &task_handle_main);
215 #else /* No os and ntshell */
216  cpu_unlock(); /* unlock cpu to let interrupt work */
217 #endif
218 
219 #endif /* MID_NTSHELL */
220 
221 #if defined(OS_FREERTOS) && defined (MID_LWIP)
222  xTaskCreate((TaskFunction_t)task_wifi, "wifi-conn", TASK_STACK_SIZE_WIFI,
223  (void *)1, TASK_PRI_WIFI, &task_handle_wifi);
224 #endif
225 
226 #ifdef OS_FREERTOS
227  vTaskStartScheduler();
228 #endif
229 
230  task_main((void *)(&s_main_args));
231 /* board level exit */
232 }
