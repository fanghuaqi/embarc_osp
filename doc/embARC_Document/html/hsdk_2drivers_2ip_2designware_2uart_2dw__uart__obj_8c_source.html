372 DEV_UART_PTR uart_get_dev(int32_t uart_id)
373 {
374  static uint32_t install_flag = 0;
375 
376  /* intall device objects */
377  if (install_flag == 0) {
378  install_flag = 1;
379  dw_uart_all_install();
380  }
381 
382  switch (uart_id) {
383 #if (USE_DW_UART_0)
384  case DW_UART_0_ID:
385  return &dw_uart_0;
386  break;
387 #endif
388 #if (USE_DW_UART_1)
389  case DW_UART_1_ID:
390  return &dw_uart_1;
391  break;
392 #endif
393 #if (USE_DW_UART_2)
394  case DW_UART_2_ID:
395  return &dw_uart_2;
396  break;
397 #endif
398 #if (USE_DW_UART_3)
399  case DW_UART_3_ID:
400  return &dw_uart_3;
401  break;
402 #endif
403  default:
404  break;
405  }
406  return NULL;
407 }
