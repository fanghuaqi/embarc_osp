39 int32_t cy8c95xx_reg_write(CY8C95XX_DEF_PTR obj, uint8_t dev, uint8_t regaddr, uint8_t *val, uint8_t len)
40 {
41  int32_t ercd = E_OK;
42  uint8_t data[1];
43  uint32_t slvaddr;
44  DEV_IIC_PTR iic_obj = iic_get_dev(obj->i2c_id);
45 
46  dbg_printf(DBG_LESS_INFO, "[%s]%d: obj 0x%x, regaddr 0x%x, val 0x%x\r\n", __FUNCTION__, __LINE__, obj, regaddr, *val);
47  dbg_printf(DBG_MORE_INFO, "[%s]%d: iic_obj 0x%x -> 0x%x\r\n", __FUNCTION__, __LINE__, iic_obj, *iic_obj);
48  CY8C95XX_CHECK_EXP_NORTN(iic_obj != NULL, E_PAR);
49  CY8C95XX_CHECK_EXP_NORTN((dev == 0) || (dev == 1), E_PAR);
50 
51  if (dev == 0) {
52  slvaddr = obj->slvaddr_io;
53  } else {
54  slvaddr = obj->slvaddr_eep;
55  }
56 
57  data[0] = (uint8_t)(regaddr & 0xff);
59  ercd = iic_obj->iic_control(IIC_CMD_MST_SET_TAR_ADDR, CONV2VOID(slvaddr));
60  ercd = iic_obj->iic_control(IIC_CMD_MST_SET_NEXT_COND, CONV2VOID(IIC_MODE_RESTART));
61  ercd = iic_obj->iic_write(data, 1);
62  ercd = iic_obj->iic_control(IIC_CMD_MST_SET_NEXT_COND, CONV2VOID(IIC_MODE_STOP));
63  ercd = iic_obj->iic_write(val, len);
64 
65 error_exit:
66  return ercd;
67 }
