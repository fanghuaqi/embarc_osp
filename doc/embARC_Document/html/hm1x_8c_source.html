132 void hm1x_flush(HM1X_DEF_PTR obj)
133 {
134  uint32_t rd_cnt = 0;
135  uint8_t read_chr[MAX_READ_CNT];
136  uint32_t rd_avail;
137  DEV_UART *uart_obj = uart_get_dev(obj->uart_id);
138 
139  uart_obj->uart_control(UART_CMD_GET_RXAVAIL, (void *)(&rd_avail));
140  do {
141  rd_cnt = (rd_avail>MAX_READ_CNT)?MAX_READ_CNT:rd_avail;
142  if (rd_avail > 0) {
143  uart_obj->uart_read((void *)read_chr, rd_cnt);
144  }
145  uart_obj->uart_control(UART_CMD_GET_RXAVAIL, (void *)(&rd_avail));
146  } while(rd_avail > 0);
147 }

233 int32_t hm1x_exec_command(HM1X_DEF_PTR obj, char *cmd, char *resp, uint32_t wait_ms)
234 {
235  uint32_t cur_ofs = 0;
236  uint32_t read_cnt;
237  uint32_t cur_time;
238  if ((cmd == NULL) || (resp == NULL)) {
239  return E_PAR;
240  }
241 
242  if (hm1x_write(obj, (uint8_t *)cmd, strlen(cmd)) > 0) {
243  cur_time = HM1X_TIMENOW();
244  do {
245  read_cnt = hm1x_read(obj, (uint8_t *)(&resp[cur_ofs]), 2);
246  cur_ofs += read_cnt;
247  } while((HM1X_TIMENOW()-cur_time) < wait_ms);
248  resp[cur_ofs] = '\0';
249  return cur_ofs;
250  }
251  return -1;
252 }

263 int32_t hm1x_exec_cmd_chkresp(HM1X_DEF_PTR obj, char *cmd, char *resp_prefix, char *resp_data, uint32_t max_try)
264 {
265  char resp[MAX_RESP_LENGTH];
266  uint32_t try_cnt = 0;
267  int32_t rd_cnt;
268  int32_t resp_prefix_len = 0;
269  int32_t resp_data_len = -1;
270 
271  if (cmd == NULL) {
272  return E_PAR;
273  }
274 
275  if (resp_prefix != NULL) {
276  resp_prefix_len = strlen(resp_prefix);
277  }
278 
279  do {
280  rd_cnt = hm1x_exec_command(obj, cmd, resp, HM1X_READ_TMOUT);
281  if (rd_cnt >= resp_prefix_len) {
282  if (resp_prefix_len > 0) {
283  if (strncmp(resp, resp_prefix, resp_prefix_len) != 0) {
284  continue;
285  }
286  }
287  if (resp_data == NULL) {
288  resp_data_len = 0;
289  } else {
290  resp_data_len = rd_cnt - resp_prefix_len;
291  if (resp_data_len > 0) {
292  strcpy(resp_data, &resp[resp_prefix_len]);
293  }
294  }
295  break;
296  }
297  } while (try_cnt++ < max_try);
298 
299  return resp_data_len;
300 }

319 
320  if (hm1x_exec_cmd_chkresp(obj, cmd_buf, OK_SET_PREFIX, NULL, 2) >=0 ) {
321  return E_OK;

323  return -1;
