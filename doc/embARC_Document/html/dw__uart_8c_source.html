221 static int32_t dw_uart_set_dps(DW_UART_REG *uart_reg_ptr, UART_DPS_FORMAT *dps)
222 {
223  uint32_t dps_value = 0;
224 
225  if (dps == NULL) return -1;
226  /* data bits check */
227  if ((dps->databits < 5) || (dps->databits > 8)) return -1;
228  /* stop bits check */
229  if (dps->stopbits > UART_STPBITS_TWO) return -1;
230  /* parity bit type check */
231  if (dps->parity > UART_PARITY_SPACE) return -1;
232 
233  dps_value |= (uint32_t)dw_uart_databits[dps->databits-5];
234  dps_value |= (uint32_t)dw_uart_stopbits[dps->stopbits];
235  dps_value |= (uint32_t)dw_uart_parity[dps->parity];
236 
237  /* clear dps bits */
238  uart_reg_ptr->LCR &= (~DW_UART_LCR_DPS_MASK);
239  /* set dps bits */
240  uart_reg_ptr->LCR |= dps_value;
241 
242  return 0;
243 }

323 static void dw_uart_flush_output(DEV_UART_INFO *uart_info_ptr)
324 {
325  uint32_t i;
326  char *p_charbuf;
327 
328  DW_UART_CTRL *uart_ctrl_ptr = (DW_UART_CTRL_PTR)(uart_info_ptr->uart_ctrl);
329  DW_UART_REG *uart_reg_ptr = (DW_UART_REG_PTR)(uart_ctrl_ptr->dw_uart_regbase);
330 
331  if (uart_info_ptr->tx_buf.buf != NULL) {
332  p_charbuf = (char *)(uart_info_ptr->tx_buf.buf);
333  for (i = uart_info_ptr->tx_buf.ofs; i < uart_info_ptr->tx_buf.len; i ++) {
334  dw_uart_psnd_chr(uart_reg_ptr, p_charbuf[i]);
335  }
336  /* clear transmit buffer */
337  uart_info_ptr->tx_buf.buf = NULL;
338  uart_info_ptr->tx_buf.len = 0;
339  uart_info_ptr->tx_buf.ofs = 0;
340  }
341  if (uart_reg_ptr->CPR & DW_UART_CPR_FIFO_STAT) {
342  /* wait until transmit fifo is empty */
343  while ((uart_reg_ptr->USR & DW_UART_USR_TFE) == 0);
344  }
345  while (uart_reg_ptr->USR & DW_UART_USR_BUSY);
346 }

353 static void dw_uart_dis_cbr(DEV_UART_INFO *uart_info_ptr, uint32_t cbrtn)
354 {
355  DW_UART_CTRL *uart_ctrl_ptr = (DW_UART_CTRL_PTR)(uart_info_ptr->uart_ctrl);
356  DW_UART_REG *uart_reg_ptr = (DW_UART_REG_PTR)(uart_ctrl_ptr->dw_uart_regbase);
357 
358  switch (cbrtn) {
359  case DW_UART_RDY_SND:
360  uart_reg_ptr->IER &= ~DW_UART_IER_XMIT_EMPTY;
361  uart_ctrl_ptr->int_status &= ~DW_UART_TXINT_ENABLE;
362  break;
363  case DW_UART_RDY_RCV:
364  uart_reg_ptr->IER &= ~DW_UART_IER_DATA_AVAIL;
365  uart_ctrl_ptr->int_status &= ~DW_UART_RXINT_ENABLE;
366  break;
367  default:
368  break;
369  }
370  if (uart_ctrl_ptr->int_status & DW_UART_GINT_ENABLE) {
371  if ((uart_ctrl_ptr->int_status & (DW_UART_RXINT_ENABLE|DW_UART_TXINT_ENABLE)) == 0) {
372  if (uart_ctrl_ptr->intno != DW_UART_INVALID_INTNO) {
373  int_disable(uart_ctrl_ptr->intno);
374  }
375  uart_ctrl_ptr->int_status &= ~DW_UART_GINT_ENABLE;
376  }
377  }
378 }

551 int32_t dw_uart_open (DEV_UART *uart_obj, uint32_t baud)
552 {
553  int32_t ercd = E_OK;
554  DEV_UART_INFO *uart_info_ptr = &(uart_obj->uart_info);
555 
556  /* START ERROR CHECK */
557  VALID_CHK_UART_INFO_OBJECT(uart_info_ptr);
558  DW_UART_CHECK_EXP(baud>0, E_PAR);
559  /* END OF ERROR CHECK */
560 
561  uart_info_ptr->opn_cnt ++;
562  if (uart_info_ptr->opn_cnt > 1) { /* opened before */
563  if (baud == uart_info_ptr->baudrate) { /* baudrate is the same */
564  return E_OK;
565  } else { /* open with different baudrate */
566  return E_OPNED;
567  }
568  }
569 
570  int32_t baud_divisor = 0;
571 
572  DW_UART_CTRL *uart_ctrl_ptr = (DW_UART_CTRL_PTR)(uart_info_ptr->uart_ctrl);
573  DW_UART_REG *uart_reg_ptr = (DW_UART_REG_PTR)(uart_ctrl_ptr->dw_uart_regbase);
574 
575  /* Get FIFO Length */
576  if (uart_reg_ptr->CPR & DW_UART_CPR_FIFO_STAT) {
577  uart_ctrl_ptr->tx_fifo_len = dw_uart_get_txfifo_len(uart_reg_ptr);
578  uart_ctrl_ptr->rx_fifo_len = dw_uart_get_rxfifo_len(uart_reg_ptr);
579  } else {
580  uart_ctrl_ptr->tx_fifo_len = 0;
581  uart_ctrl_ptr->rx_fifo_len = 0;
582  }
583 
585  uart_info_ptr->baudrate = baud;
586  baud_divisor = DW_UART_BAUD2DIV(uart_ctrl_ptr->dw_apb_bus_freq, baud);
587  uart_info_ptr->dps_format = dps_format_default;
588  uart_info_ptr->hwfc = hwfc_default;
589  dw_uart_init(uart_reg_ptr, baud_divisor, &(uart_info_ptr->dps_format), uart_info_ptr->hwfc);
590 
591  uart_info_ptr->status = DEV_ENABLED;
592  uart_info_ptr->extra = NULL;
593 
597  dw_uart_disable_interrupt(uart_info_ptr);
599  if (uart_ctrl_ptr->intno != DW_UART_INVALID_INTNO) {
600  int_handler_install(uart_ctrl_ptr->intno, uart_ctrl_ptr->dw_uart_int_handler);
601  }
602 
603  memset(&(uart_info_ptr->tx_buf), 0, sizeof(DEV_BUFFER));
604  memset(&(uart_info_ptr->rx_buf), 0, sizeof(DEV_BUFFER));
605  memset(&(uart_info_ptr->uart_cbs), 0, sizeof(DEV_UART_CBS));
606 
607 error_exit:
608  return ercd;
609 }

618 int32_t dw_uart_close (DEV_UART *uart_obj)
619 {
620  int32_t ercd = E_OK;
621  DEV_UART_INFO *uart_info_ptr = &(uart_obj->uart_info);
622 
623  /* START ERROR CHECK */
624  VALID_CHK_UART_INFO_OBJECT(uart_info_ptr);
625  DW_UART_CHECK_EXP(uart_info_ptr->opn_cnt > 0, E_OK);
626  /* END OF ERROR CHECK */
627 
628  uart_info_ptr->opn_cnt --;
629  if (uart_info_ptr->opn_cnt == 0) {

632  dw_uart_abort_rx(uart_obj);

635  memset(&(uart_info_ptr->rx_buf), 0, sizeof(DEV_BUFFER));
636  memset(&(uart_info_ptr->uart_cbs), 0, sizeof(DEV_UART_CBS));
637  dw_uart_disable_device(uart_info_ptr);
638  uart_info_ptr->status = 0;
639  uart_info_ptr->extra = NULL;
640  } else {
641  ercd = E_OPNED;
642  }
643 
644 error_exit:
645  return ercd;
646 }

894 void dw_uart_isr (DEV_UART *uart_obj, void *ptr)
895 {
896  int32_t ercd = E_OK;
897  DEV_UART_INFO *uart_info_ptr = &(uart_obj->uart_info);
898 
899  /* START ERROR CHECK */
900  VALID_CHK_UART_INFO_OBJECT(uart_info_ptr);
901  /* END OF ERROR CHECK */
902 
903  uint32_t uart_int_status;
904  volatile uint32_t temp;
905  DEV_BUFFER *buf_ptr;
906  char *p_charbuf;
907 
908  DW_UART_CTRL *uart_ctrl_ptr = (DW_UART_CTRL_PTR)(uart_info_ptr->uart_ctrl);
909  DW_UART_REG *uart_reg_ptr = (DW_UART_REG_PTR)(uart_ctrl_ptr->dw_uart_regbase);
910 
912  uart_int_status = (uart_reg_ptr->IIR) & DW_UART_IIR_INT_ID_MASK;
913 
914  switch (uart_int_status) {
915  case DW_UART_IIR_MDM_STATUS:
916  temp = (volatile uint32_t)(uart_reg_ptr->MSR);
917  break;
918  case DW_UART_IIR_LINE_STATUS:
919  if (uart_info_ptr->uart_cbs.err_cb) {
920  uart_info_ptr->uart_cbs.err_cb(uart_info_ptr);
921  }
922  temp = (volatile uint32_t)(uart_reg_ptr->LSR);
923  break;
924  case DW_UART_IIR_XMIT_EMPTY:
925  buf_ptr = &(uart_info_ptr->tx_buf);
926  p_charbuf = (char *)buf_ptr->buf;
927  if (p_charbuf != NULL) {
928  while (dw_uart_putready(uart_reg_ptr)) {
929  dw_uart_putchar(uart_reg_ptr, p_charbuf[buf_ptr->ofs]);
930  buf_ptr->ofs ++;
931  if (buf_ptr->ofs >= buf_ptr->len) {
932  dw_uart_dis_cbr(uart_info_ptr, DW_UART_RDY_SND);
933  if (uart_info_ptr->uart_cbs.tx_cb) {
934  uart_info_ptr->uart_cbs.tx_cb(uart_obj);
935  }
936  /* clear the send buffer pointer */
937  memset(buf_ptr, 0, sizeof(DEV_BUFFER));
938  break;
939  }
940  }
941  } else {
942  if (uart_info_ptr->uart_cbs.tx_cb) {
943  uart_info_ptr->uart_cbs.tx_cb(uart_obj);
944  }
945  }
946  break;
947  case DW_UART_IIR_RX_TIMEOUT:
948  temp = dw_uart_getchar(uart_reg_ptr);
949  break;
950  case DW_UART_IIR_DATA_AVAIL:
951  buf_ptr = &(uart_info_ptr->rx_buf);
952  p_charbuf = (char *)buf_ptr->buf;
953  if (p_charbuf != NULL) {
954  while (dw_uart_getready(uart_reg_ptr)) {
955  p_charbuf[buf_ptr->ofs] = (char)dw_uart_getchar(uart_reg_ptr);
956  buf_ptr->ofs ++;
957  if (buf_ptr->ofs >= buf_ptr->len) {
958  dw_uart_dis_cbr(uart_info_ptr, DW_UART_RDY_RCV);
959  if (uart_info_ptr->uart_cbs.rx_cb) {
960  uart_info_ptr->uart_cbs.rx_cb(uart_obj);
961  }
962  /* clear the send buffer pointer */
963  memset(buf_ptr, 0, sizeof(DEV_BUFFER));
964  break;
965  }
966  }
967  } else {
968  if (uart_info_ptr->uart_cbs.rx_cb) {
969  uart_info_ptr->uart_cbs.rx_cb(uart_obj);
970  }
971  }
972  break;
973  default:
974  temp = (volatile uint32_t)(uart_reg_ptr->USR);
975  break;
976  }
977 
978 error_exit:
979  return;
