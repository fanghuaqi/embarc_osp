150 int32_t spi_xfer(SPI_XFER *xfer)
151 {
152  static __attribute__((aligned(16))) dma_desc_t dma_desc_tx[SPI_XFER_LIST_LEN];
153  static __attribute__((aligned(16))) dma_desc_t dma_desc_rx[SPI_XFER_LIST_LEN];
154  static dma_ctrl_t dma_ctrl_tx, dma_ctrl_rx;
155  dma_callback_t xfer_cb = NULL;
156 
157  SPI_XFER *cur_xfer = xfer;
158  uint8_t i = 0;
159  uint8_t j = 0;
160  uint32_t txtemp = 0xFFFFFFFF, rxtemp = 0;
161 
162  while (cur_xfer) {
163  DMA_CTRL_SET_OP(&dma_ctrl_tx, DMA_MANUAL_LINKED_TRANSFER);
164  DMA_CTRL_SET_RT(&dma_ctrl_tx, DMA_MANUAL_REQUEST);
165  DMA_CTRL_SET_DTT(&dma_ctrl_tx, DMA_MEM2MEM);
166  DMA_CTRL_SET_DWINC(&dma_ctrl_tx, DMA_DW1INC1);
167  DMA_CTRL_SET_ARB(&dma_ctrl_tx, SPI_XFER_ARB_SZ);
168  DMA_CTRL_SET_INT(&dma_ctrl_tx, DMA_INT_DISABLE);
169  if (cur_xfer->tx_buf) {
170  DMA_CTRL_SET_AM(&dma_ctrl_tx, DMA_AM_SRCINC_DSTNOT);
171  dmac_config_desc(&dma_desc_tx[i], (void *)(cur_xfer->tx_buf), (void *)(&spi_reg->DATAREG), cur_xfer->len, &dma_ctrl_tx);
172  } else {
173  DMA_CTRL_SET_AM(&dma_ctrl_tx, DMA_AM_SRCNOT_DSTNOT);
174  dmac_config_desc(&dma_desc_tx[i], (void *)(&txtemp), (void *)(&spi_reg->DATAREG), cur_xfer->len, &dma_ctrl_tx);
175  }
176  DMA_CTRL_SET_OP(&dma_ctrl_rx, DMA_MANUAL_LINKED_TRANSFER);
177  DMA_CTRL_SET_RT(&dma_ctrl_rx, DMA_MANUAL_REQUEST);
178  DMA_CTRL_SET_DTT(&dma_ctrl_rx, DMA_MEM2MEM);
179  DMA_CTRL_SET_DWINC(&dma_ctrl_rx, DMA_DW1INC1);
180  DMA_CTRL_SET_ARB(&dma_ctrl_rx, SPI_XFER_ARB_SZ);
181  DMA_CTRL_SET_INT(&dma_ctrl_rx, DMA_INT_DISABLE);
182  if (cur_xfer->rx_buf) {
183  DMA_CTRL_SET_AM(&dma_ctrl_rx, DMA_AM_SRCNOT_DSTINC);
184  dmac_config_desc(&dma_desc_rx[i], (void *)(&spi_reg->DATAREG), (void *)(cur_xfer->rx_buf), cur_xfer->len, &dma_ctrl_rx);
185  } else {
186  DMA_CTRL_SET_AM(&dma_ctrl_rx, DMA_AM_SRCNOT_DSTNOT);
187  dmac_config_desc(&dma_desc_rx[i], (void *)(&spi_reg->DATAREG), (void *)(&rxtemp), cur_xfer->len, &dma_ctrl_rx);
188  }
189  i ++;
190  if (i >= SPI_XFER_LIST_LEN) {
191  break;
192  }
193  cur_xfer = cur_xfer->next;
194  }
195  dmac_desc_add_linked(&dma_desc_tx[i-1], NULL);
196  dmac_desc_add_linked(&dma_desc_rx[i-1], NULL);
197  for (j = i; j > 1; j --) {
198  dmac_desc_add_linked(&dma_desc_tx[j-2], &dma_desc_tx[j-1]);
199  dmac_desc_add_linked(&dma_desc_rx[j-2], &dma_desc_rx[j-1]);
200  }
201  /* Init and configure dma channel transfer with transfer descriptor */
202  dmac_config_channel(&dma_chn_tx, &dma_desc_tx[0]);
203  dmac_config_channel(&dma_chn_rx, &dma_desc_rx[0]);
204 
205  dmac_stop_channel(&dma_chn_rx);
206  dmac_stop_channel(&dma_chn_tx);
207 
208 #ifdef SPIFLASH_ENABLE_UDMA_INTERRUPT
209  xfer_cb = spi_xfer_callback;
210 #else
211  arc_lock();
212  xfer_cb = NULL;
213 #endif
214  /* select device */
215  spi_reg->SSIENR = DW_SPI_SSI_DISABLE;
216  spi_reg->DMACR = 0;
217  spi_reg->DMATDLR = 32-SPI_XFER_ARB_SZ;
218  spi_reg->DMARDLR = SPI_XFER_ARB_SZ-1;
219  flush_xfer_data(xfer);
220  dmac_start_channel(&dma_chn_rx, xfer_cb, DMA_CHN_HIGH_PRIO);
221  dmac_start_channel(&dma_chn_tx, xfer_cb, DMA_CHN_HIGH_PRIO);
222  spi_reg->SER = 1<<SPIFLASH_LINE_CS;
223  spi_reg->SSIENR = DW_SPI_SSI_ENABLE;
224  /* enable rx and tx dma */
225  spi_reg->DMACR = 3;
226  dmac_wait_channel(&dma_chn_tx);
227  dmac_wait_channel(&dma_chn_rx);
228  /* deselect device */
229  spi_reg->SSIENR = DW_SPI_SSI_DISABLE;
230  spi_reg->DMACR = 0;
231  spi_reg->DMATDLR = 0;
232  spi_reg->DMARDLR = 0;
233  spi_reg->SER = 0;
234  spi_reg->SSIENR = DW_SPI_SSI_ENABLE;
235 #ifndef SPIFLASH_ENABLE_UDMA_INTERRUPT
236  arc_unlock();
237 #endif
238  invalidate_xfer_data(xfer);
239 
240  return 0;
241 }
