100 static int32_t ntshell_uart_nt_read(NTSHELL_IO *nt_io, void *buf, uint32_t cnt)
101 {
102  int32_t ercd = E_OK;
103  DEV_UART *uart_dev;
104  NTSHELL_IO_UART *nt_uart;
105  int32_t rd_avail = 0;
106 
107  EMSK_NTSHELL_UART_CHECK_EXP(nt_io!=NULL, E_OBJ);
108  EMSK_NTSHELL_UART_CHECK_EXP(nt_io->extra_info!=NULL, E_OBJ);
109 
110  nt_uart = (NTSHELL_IO_UART *)nt_io->extra_info;
111  uart_dev = uart_get_dev(nt_uart->uart_id);
112 
113  EMSK_NTSHELL_UART_CHECK_EXP(uart_dev!=NULL, E_PAR);
114 
115  ercd = cnt;
116  do {
117  uart_dev->uart_control(UART_CMD_GET_RXAVAIL, (void *)(&rd_avail));
118  if (rd_avail > 0) {
119  rd_avail = (rd_avail > cnt) ? cnt : rd_avail;
120  uart_dev->uart_read(buf, rd_avail);
121  buf = (void *)(((uint32_t)buf) + rd_avail);
122  cnt -= rd_avail;
123  } else {
124 #ifdef OS_FREERTOS
125  vTaskDelay(5);
126 #endif
127  }
128  } while (cnt > 0);
129 
130 
131 error_exit:
132  return ercd;
133 }

155 static void ntshell_uart_nt_printf(NTSHELL_IO *nt_io, const char *fmt, va_list arp)
156 {
157  int32_t ercd = E_OK;
158  NTSHELL_IO_UART *nt_uart;
159  void (*pf)(unsigned char);
160 
161  EMSK_NTSHELL_UART_CHECK_EXP(nt_io!=NULL, E_OBJ);
162  EMSK_NTSHELL_UART_CHECK_EXP(nt_io->extra_info!=NULL, E_OBJ);
163 
164  nt_uart = (NTSHELL_IO_UART *)nt_io->extra_info;
165 
166  if (nt_uart->write_byte) {
167  pf = xfunc_out; /* Save current output device */
168  xfunc_out = nt_uart->write_byte; /* Switch output to specified device */
169  xvprintf(fmt, arp);
170  xfunc_out = pf; /* Restore output device */
171  }
172 
173 error_exit:
174  return ;
175 }
