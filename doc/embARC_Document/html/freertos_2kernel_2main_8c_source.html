144 static void perf_start(void)

152 static unsigned int perf_end(void)
153 {
154  unsigned int end = 0;
155 
156  if (timer_current(perf_id, (void *)(&end)) < 0) {
157  return 0;
158  }
159 
160  if (start < end) {
161  return (end - start);
162  } else {
163  return (0xFFFFFFFF - start + end);
164  }
165 }

170 int main(void)
171 {
172  EMBARC_PRINTF("Benchmark CPU Frequency: %d Hz\r\n", BOARD_CPU_CLOCK);
173  EMBARC_PRINTF("Benchmark will run %d times, please wait about %d ms\r\n", MAX_SAMPLES, (TASK_DELAY_MS*MAX_SAMPLES));

175  // Register interrupts

179 
180  exc_handler_install(EXC_NO_TRAP, trap_exception);
182  vTaskSuspendAll();
183 
184  // Create Tasks
185  if (xTaskCreate(task1, "task1", 128, (void *)1, TSK_PRIOR_LO, &task1_handle)
186  != pdPASS) {
187  EMBARC_PRINTF("create task1 error\r\n");
188  return -1;
189  }
190 
191  if (xTaskCreate(task2, "task2", 128, (void *)2, TSK_PRIOR_HI, &task2_handle)
192  != pdPASS) {
193  EMBARC_PRINTF("create task2 error\r\n");
194  return -1;
195  }
196 
197  // Create Mutex
198  mux1_id = xSemaphoreCreateMutex();
199 
200  // Create Semaphores
201  sem1_id = xSemaphoreCreateBinary();
202  xSemaphoreGive(sem1_id);
203 
204  // Create Events
205  evt1_cb = xEventGroupCreate();
206 
207  // Create Queues
208  dtq1_id = xQueueCreate(1, sizeof(uint32_t));
209 
210  xTaskResumeAll();
211 
212  vTaskSuspend(NULL);
213  vTaskSuspend(NULL);
214 
215  return 0;
216 }

224 static void task1(void *par)
225 {
226  uint32_t queue_data = 1;
227  TickType_t xLastExecutionTime;
228  xLastExecutionTime = xTaskGetTickCount();
230  while (1) {
232  x.t_t2_t1 = perf_end();
234  perf_start();
235  _arc_aux_write(AUX_IRQ_HINT, SWI_INTNO);
236  x.t_int_t1 = perf_end();
238 
239  // task delay, to control benchmark run speed
240  vTaskDelayUntil( &xLastExecutionTime, TASK_DELAY_MS);
241  // Task 1 acquire mutex first
242  xSemaphoreTake(mux1_id, portMAX_DELAY);
243  // Task 1 acquire semaphore first
244  xSemaphoreTake(sem1_id, portMAX_DELAY);
245 
247  perf_start();
248  vTaskResume(task2_handle);
249 
251  perf_start();
252  // Task 1 release mutex, task 2 acquire it
253  xSemaphoreGive(mux1_id);
254 
256  perf_start();
257  // Task 1 post sem, task 2 acquire it
258  xSemaphoreGive(sem1_id);
259 
261  perf_start();
262  // Task 1 write event, task 2 read it
263  xEventGroupSetBits(evt1_cb, EVENT_WAIT_BITS);
264 
266  perf_start();
267  // Task 1 write queue, task 2 read it
268  xQueueSend(dtq1_id, (void *)(&queue_data), portMAX_DELAY);
269  queue_data ++;
270  }
