121 int32_t dma_prepare(void)
122 {
123  DMA_CTRL_SET_OP(&dma_ctrl, DMA_SINGLE_TRANSFER);
124  DMA_CTRL_SET_RT(&dma_ctrl, DMA_AUTO_REQUEST);
125  DMA_CTRL_SET_DTT(&dma_ctrl, DMA_MEM2MEM);
126  DMA_CTRL_SET_DWINC(&dma_ctrl, DMA_DW1INC1);
127  DMA_CTRL_SET_AM(&dma_ctrl, DMA_AM_SRCINC_DSTINC);
128  DMA_CTRL_SET_ARB(&dma_ctrl, 255);
129  DMA_CTRL_SET_INT(&dma_ctrl, DMA_INT_DISABLE);
130 
131  dmac_config_desc(&dma_desc, NULL, NULL, 0, &dma_ctrl);
132 
133  dmac_desc_add_linked(&dma_desc, NULL);
134  /* Init and configure dma channel transfer with transfer descriptor */
135  dmac_init_channel(&dma_chn);
136  dmac_config_channel(&dma_chn, &dma_desc);
137 
138  /* Reserve a channel for use */
139  dma_channel = dmac_reserve_channel(DMA_CHN_ANY, &dma_chn, DMA_REQ_SOFT);
140  return dma_channel;
141 }

172  // _ICACHE_INVALIDATE_MLINES((void *)src, size);

177  dmac_start_channel(&dma_chn, xfer_cb, DMA_CHN_HIGH_PRIO);
178  cycles_cnt[cycles_idx++] = perf_end();

182  ercd = -1;

184  ercd = 0;

186  cycles_cnt[cycles_idx++] = perf_end();

192  // _ICACHE_INVALIDATE_MLINES((void *)dst, size);
193  cycles_cnt[cycles_idx++] = perf_end();
