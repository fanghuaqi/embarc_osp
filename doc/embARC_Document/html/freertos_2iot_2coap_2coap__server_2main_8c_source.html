128 static void hnd_get_temp(coap_context_t *ctx, struct coap_resource_t *resource,
129  coap_address_t *peer, coap_pdu_t *request, str *token,
130  coap_pdu_t *response) {
131 
132  coap_opt_iterator_t opt_iter;
133  unsigned char buf[40];
134  size_t len;
135  float temp_val;
136  coap_subscription_t *subscription;
137 
138  response->hdr->code = COAP_RESPONSE_CODE(205);
139 
140  /* handle observe request */
141  if (request != NULL &&
142  coap_check_option(request, COAP_OPTION_OBSERVE, &opt_iter)) {
143  subscription = coap_add_observer(resource, peer, token);
144  if (subscription) {
145  subscription->non = request->hdr->type == COAP_MESSAGE_NON;
146  coap_add_option(response, COAP_OPTION_OBSERVE, 0, NULL);
147  }
148  }
149 
150  if (resource->dirty == 1) {
151  coap_add_option(response, COAP_OPTION_OBSERVE,
152  coap_encode_var_bytes(buf, ctx->observe), buf);
153  }
154 
155  coap_add_option(response, COAP_OPTION_CONTENT_FORMAT,
156  coap_encode_var_bytes(buf, COAP_MEDIATYPE_TEXT_PLAIN), buf);
157 
158  /* temp data is fresh in 60 seconds */
159  coap_add_option(response, COAP_OPTION_MAXAGE,
160  coap_encode_var_bytes(buf, 60), buf);
161 
162  if (adt7420_sensor_read(temp, &temp_val) != E_OK) {
163  temp_val = 0.0f;
164  }
165 
166  len = snprintf((char *)buf,
167  min(sizeof(buf), response->max_size - response->length),
168  "%f", temp_val);
169  coap_add_data(response, len, buf);
170 }

228 static void task_coap_server(void *par)
229 {
230  coap_context_t *ctx;
231  coap_address_t addr;
232 
233  addr.port = 5683;
234  addr.addr = netif_default->ip_addr;
235 
236  ctx = coap_new_context(&addr);
237 
238  if (!ctx) {
239  EMBARC_PRINTF("create coap context error, run failed!\r\n");
240  vTaskDelete(NULL);
241  return;
242  }
243 
244  init_resources(ctx);
245 
246  if (xTaskCreate(task_temp_tx, "temperature update task", 128, (void *)1,
247  TSKPRI_TEMPTX, &task_temptx_handle) != pdPASS) {
248  EMBARC_PRINTF("create temperature update task failed\r\n");
249  vTaskDelete(NULL);
250  return;
251  }
257  while (1) {
258  coap_dispatch(ctx);
259  coap_check_notify(ctx);
260  vTaskDelay(100);
261  }
262  coap_free_context(ctx);
263 }
