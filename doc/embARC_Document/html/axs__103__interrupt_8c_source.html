246 int32_t int_disable(const uint32_t intno)
247 {
248  uint32_t ictl = intno >> 8;
249 
250  uint32_t sub_intno = intno & 0xff;
251 
252  if (ictl == 0 && sub_intno >= NUM_EXC_CPU && sub_intno < NUM_EXC_ALL) {
253  /* arc interrupt controller */
254  arc_int_disable(sub_intno);
255  } else if (ictl == INTNO_ICTL_MB && sub_intno <= ICTL_INTNO_NUM ) { /* main board ictl */
256  dw_ictl_int_disable(dw_ictl_ptr, sub_intno);
257  } else if (ictl == INTNO_ICTL_CPU && sub_intno <= CPU_ICTL_INTNO_NUM) { /* cpu card ictl */
258  ictl_gpio_int_disable(ictl_gpio_ptr, sub_intno);
259  } else {
260  return -1;
261  }
262 
263  return 0;
264 }

564

568  int_cfg.int_bit_debounce = GPIO_INT_BITS_DIS_DEBOUNCE(ICTL_GPIOA_LINE_MASK);
569  port_ptr->gpio_control(GPIO_CMD_SET_BIT_INT_CFG, (void *)(&int_cfg));
570  bit_isr.int_bit_ofs = ICTL_GPIOA_LINE;
571  bit_isr.int_bit_handler = ictl_gpio_int_isr;
572  port_ptr->gpio_control(GPIO_CMD_SET_BIT_ISR, (void *)(&bit_isr));

575  int_handler_install(INTNO_ICTL_CPU, dw_ictl_int_isr);
576  int_enable(INTNO_ICTL_CPU);
577 }
