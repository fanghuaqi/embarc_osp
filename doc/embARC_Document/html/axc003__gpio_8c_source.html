90 void leda_write(uint32_t led_val, uint32_t mask)
91 {
92  AXC003_GPIO_CHECK_EXP_NORTN(leda_port != NULL);
93 
94  led_val &= mask;
95  led_val = (led_val & 0x3) | ((led_val & 0xC) << 0x3) | ((led_val & 0x30) << 0x6) |
96  ((led_val & 0xC0) << 9);
97 
98  mask = (mask & 0x3) | ((mask & 0xC) << 0x3) | ((mask & 0x30) << 0x6) |
99  ((mask & 0xC0) << 9);
100 
101  leda_port->gpio_write(led_val, mask);
102 
103 error_exit:
104  return;
105 }

124 uint32_t button_read(uint32_t mask)
125 {
126  uint32_t value;
127 
128  AXC003_GPIO_CHECK_EXP_NORTN(button_port != NULL);
129 
130  mask = (mask << AXC003_BUTTON_OFFSET) & AXC003_BUTTON_MASK;
131  button_port->gpio_read(&value, mask);
132  value = (~value) & mask;
133 
134  return (value >> AXC003_BUTTON_OFFSET);
135 
136 error_exit:
137  return 0;
138 }

140 void button_install_isr(uint32_t offset, void (*isr_ptr)(void *ptr))
141 {
142 
143 /* note: CPU card's creg ctrl's gpio int source to INTNO_ICTL_MB,
144 default is A12 which is int request from main board, int request from SW2504-SW2507 can not
145 go to INTNO_ICTL_MB */
146  DEV_GPIO_BIT_ISR bit_isr;
147  DEV_GPIO_INT_CFG int_cfg;
148  uint32_t mask;
149 
150  if (offset < AXC003_BUTTON_SW2504_OFFSET || offset > AXC003_BUTTON_SW2507_OFFSET) {
151  return;
152  }
153  AXC003_GPIO_CHECK_EXP_NORTN(button_port != NULL);
154 
155  mask = (0x1 << offset);
156 
157  int_cfg.int_bit_mask = mask;
158  int_cfg.int_bit_type = GPIO_INT_BITS_EDGE_TRIG(mask);
159  int_cfg.int_bit_polarity = GPIO_INT_BITS_POL_FALL_EDGE(mask);
160  int_cfg.int_bit_debounce = GPIO_INT_BITS_ENA_DEBOUNCE(mask);
161  button_port->gpio_control(GPIO_CMD_SET_BIT_INT_CFG, (void *)(&int_cfg));
162  bit_isr.int_bit_ofs = offset;
163  bit_isr.int_bit_handler = isr_ptr;
164  button_port->gpio_control(GPIO_CMD_SET_BIT_ISR, (void *)(&bit_isr));
165  button_port->gpio_control(GPIO_CMD_ENA_BIT_INT, (void *)mask);
166 
167 error_exit:
168  return;
169 }

184 void seven_seg_write(uint32_t val)
185 {
186  uint32_t digit;
187 
188  AXC003_GPIO_CHECK_EXP_NORTN(seven_seg_port != NULL);
189 
190  if (val > 99) {
191  return; // only can show val < 99, two 7-seg display.
192  }
193 
194  digit = num_to_7seg_table[val / 10] | (num_to_7seg_table[val % 10] << 8);
195 
196  seven_seg_port->gpio_write((digit << AXC003_7SEG_OFFSET), AXC003_7SEG_MASK);
197 
198 error_exit:
199  return;
200 }
