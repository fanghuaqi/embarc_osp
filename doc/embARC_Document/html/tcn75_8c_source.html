81 {
82  int32_t ercd = E_PAR;

86  dbg_printf(DBG_LESS_INFO, "[%s]%d: obj 0x%x, regaddr 0x%x, val 0x%x\r\n", __FUNCTION__, __LINE__, obj, regaddr, *val);

88  TCN75_CHECK_EXP_NORTN(iic_obj!=NULL);
89 
90  data[0] = (uint8_t)(regaddr & 0xff);
91 
93  iic_obj->iic_control(IIC_CMD_MST_SET_TAR_ADDR, CONV2VOID(obj->slvaddr));

96  ercd = iic_obj->iic_write(data, 1);
97  ercd = iic_obj->iic_control(IIC_CMD_MST_SET_NEXT_COND, CONV2VOID(IIC_MODE_STOP));
98  ercd = iic_obj->iic_write(val, len);

100 error_exit:

186 int32_t tcn75_sensor_mode(TCN75_DEF_PTR obj, int32_t mode){
187  uint8_t config;
188  obj->op_mode = mode;
189 
190  _tcn75_reg_read(obj, TCN75_REG_CONFIG, &config, 1);
192  config &= ~TCN75_CONFIG_OP_MODE;
193  config |= (obj->op_mode == TCN75_OP_MODE_SHUTDOWN) ? TCN75_CONFIG_OP_MODE : 0x00 ;
194  _tcn75_reg_write(obj, TCN75_REG_CONFIG, &config, 1);
195  return E_OK;
196 }
