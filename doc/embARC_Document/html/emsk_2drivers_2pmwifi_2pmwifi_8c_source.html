110 static int32_t pmwifi_0_gpio_open(uint32_t pin_mask, uint32_t pin_dir)
111 {
112  int32_t ercd = E_OK;
113  uint32_t mask;
114  DEV_GPIO *pmwifi_gpio_ptr;
115 
116  pmwifi_gpio_ptr = gpio_get_dev(EMSK_PMWIFI_0_GPIO_ID);
117  PMWIFI_CHECK_EXP(pmwifi_gpio_ptr != NULL, E_OBJ);
118 
119  pin_mask = ((pin_mask & WF_GPIO_PIN_MASK_ALL)<<EMSK_PMWIFI_0_PIN_GAP);
120  mask = (pin_dir<<EMSK_PMWIFI_0_PIN_GAP) & pin_mask;
121  ercd = pmwifi_gpio_ptr->gpio_open(mask);
122  if (ercd == E_OPNED) {
123  /* direction */
124  pmwifi_gpio_ptr->gpio_control(GPIO_CMD_SET_BIT_DIR_OUTPUT, (void *)mask);
125  mask = (~(pin_dir<<EMSK_PMWIFI_0_PIN_GAP)) & pin_mask;
126  pmwifi_gpio_ptr->gpio_control(GPIO_CMD_SET_BIT_DIR_INPUT, (void *)mask);
127  }
128  /* Default all set to pull */
129  ercd = pmwifi_gpio_ptr->gpio_control(GPIO_CMD_DIS_BIT_INT, (void *)pin_mask);
130 
131 error_exit:
132  return ercd;
133 }

135 static int32_t pmwifi_0_gpio_close(uint32_t pin_mask)
136 {
137  static uint32_t close_mask = 0;
138  int32_t ercd = E_OK;
139  DEV_GPIO *pmwifi_gpio_ptr;
140 
141  pmwifi_gpio_ptr = gpio_get_dev(EMSK_PMWIFI_0_GPIO_ID);
142  PMWIFI_CHECK_EXP(pmwifi_gpio_ptr != NULL, E_OBJ);
143 
144  close_mask |= pin_mask;
145  close_mask &= WF_GPIO_PIN_MASK_ALL;
146 
147  if (close_mask == WF_GPIO_PIN_MASK_ALL) { /* only close it when all pin need to be closed */
148  ercd = pmwifi_gpio_ptr->gpio_close();
149  close_mask = 0;
150  }
151 
152 error_exit:
153  return ercd;
154 }

221  mask = (1 << EMSK_PMWIFI_0_INT_PIN);
222  ercd = pmwifi_gpio_ptr->gpio_control(GPIO_CMD_GET_BIT_MTHD, (void *)(&mthd));

226  }
