71 void hsdk_led_init(void)
72 {
73  int32_t ercd = E_OK;
74  int32_t rv = 0;
75  uint8_t buffer_tx[1];
76  uint8_t val = 0;
77 
78  cy8c95xx_obj.i2c_id = HSDK_I2C_ID_CY8C95XX;
79  cy8c95xx_obj.slvaddr_io = HSDK_I2C_ADDR_CY8C95XX_IO;
80  cy8c95xx_obj.slvaddr_eep = HSDK_I2C_ADDR_CY8C95XX_EEP;
81 
82  DEV_IIC_PTR iic_obj = iic_get_dev(cy8c95xx_obj.i2c_id);
83  ercd = iic_obj->iic_open(DEV_MASTER_MODE, IIC_SPEED_STANDARD);
84 
85  if ((ercd == E_OK) || (ercd == E_OPNED)) {
86  iic_obj->iic_control(IIC_CMD_MST_SET_TAR_ADDR, CONV2VOID(cy8c95xx_obj.slvaddr_io));
87  }
88 
89 
90  /* Select port */
91  buffer_tx[0] = CY8C95XX_PORT_1;
92  rv = cy8c95xx_reg_write(&cy8c95xx_obj, CY8C95XX_DEV_IO, CY8C95XX_PORT_SELECT_REG, buffer_tx, 1);
93  cy8c95xx_delay_us(100);
94 
95  do {
96  cy8c95xx_reg_read(&cy8c95xx_obj, CY8C95XX_DEV_IO, CY8C95XX_PORT_SELECT_REG, &val, 1);
97  } while (val != buffer_tx[0]);
98  /* Set direction: all pins output */
99  buffer_tx[0] = 0x00;
100  rv = cy8c95xx_reg_write(&cy8c95xx_obj, CY8C95XX_DEV_IO, CY8C95XX_PIN_DIRECTION_REG, buffer_tx, 1);
101  cy8c95xx_delay_us(100);
102  do {
103  cy8c95xx_reg_read(&cy8c95xx_obj, CY8C95XX_DEV_IO, CY8C95XX_PIN_DIRECTION_REG, &val, 1);
104  } while (val != buffer_tx[0]);
105  cy8c95xx_delay_us(100);
106  /* Set driver mode: Pull Up */
107  cy8c95xx_reg_read(&cy8c95xx_obj, CY8C95XX_DEV_IO, CY8C95XX_DRIVE_MODE_PU_REG, &val, 1);
108  buffer_tx[0] = 0xFF;
109  rv = cy8c95xx_reg_write(&cy8c95xx_obj, CY8C95XX_DEV_IO, CY8C95XX_DRIVE_MODE_PU_REG, buffer_tx, 1);
110  cy8c95xx_delay_us(100);
111 
112 error_exit:
113  return;
114 }
