73 void emsk_led_init(void)
74 {
75  led_port = gpio_get_dev(EMSK_LED_PORT);
76 
77  EMSK_GPIO_CHECK_EXP_NORTN(led_port != NULL);
78 
79  if (led_port->gpio_open(EMSK_LED_DIR) == E_OPNED) {
80  led_port->gpio_control(GPIO_CMD_SET_BIT_DIR_OUTPUT, (void *)(EMSK_LED_MASK));

123 void led_write(uint32_t led_val, uint32_t mask)
124 {
125  EMSK_GPIO_CHECK_EXP_NORTN(led_port != NULL);
126 
127  led_val = (~led_val) & mask;
128  led_val = led_val << EMSK_LED_OFFSET;
129  mask = (mask << EMSK_LED_OFFSET) & EMSK_LED_MASK;
130  led_port->gpio_write(led_val, mask);
131 
132 error_exit:
133  return;
134 }
