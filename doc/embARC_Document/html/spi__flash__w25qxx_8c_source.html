59 Inline int32_t spi_send_cmd(W25QXX_DEF_PTR dev, DEV_SPI_TRANSFER *xfer)
60 {
61  uint32_t cpu_status;
62  DEV_SPI_PTR spi_flash;
63  int32_t ercd = 0;
64 
65  spi_flash = spi_get_dev(dev->spi_master);
66 
67  cpu_status = cpu_lock_save();
68 
69  /* select device */
70  spi_flash->spi_control(SPI_CMD_MST_SEL_DEV, CONV2VOID((uint32_t)dev->cs));
71  spi_flash->spi_control(SPI_CMD_MST_SET_FREQ, CONV2VOID(dev->spi_freq));
72  spi_flash->spi_control(SPI_CMD_SET_CLK_MODE, CONV2VOID(SPI_CLK_MODE_0));
73  ercd = spi_flash->spi_control(SPI_CMD_TRANSFER_POLLING, CONV2VOID(xfer));
74  /* deselect device */
75  spi_flash->spi_control(SPI_CMD_MST_DSEL_DEV, CONV2VOID((uint32_t)dev->cs));
76 
77  cpu_unlock_restore(cpu_status);
78  return ercd;
79 }

110 int32_t w25qxx_write_enable(W25QXX_DEF_PTR dev)
111 {
112  uint8_t local_buf[3];
113  DEV_SPI_TRANSFER cmd_xfer;
114 
115  uint32_t status = 0;
116  do {
117  local_buf[0] = WREN;
118 
119  DEV_SPI_XFER_SET_TXBUF(&cmd_xfer, local_buf, 0, 1);
120  DEV_SPI_XFER_SET_RXBUF(&cmd_xfer, NULL, 0, 0);
121  DEV_SPI_XFER_SET_NEXT(&cmd_xfer, NULL);
122 
123  if (spi_send_cmd(dev, &cmd_xfer) != 0) {
124  return -1;
125  }
126 
127  status = w25qxx_read_status(dev);
128 
129  if (status == W25QXX_NOT_VALID) {
130  return -1;
131  }
132  // clear protection bits
133  // Write Protect. and Write Enable.
134  if( (status & 0xfc) && (status & 0x02) ) {
135  local_buf[0] = WRSR; // write status
136  local_buf[1] = 0x00; // write status
137  local_buf[2] = 0x00; // write status
138 
139  DEV_SPI_XFER_SET_TXBUF(&cmd_xfer, local_buf, 0, 3);
140  DEV_SPI_XFER_SET_RXBUF(&cmd_xfer, NULL, 0, 0);
141  DEV_SPI_XFER_SET_NEXT(&cmd_xfer, NULL);
142 
143  if (spi_send_cmd(dev, &cmd_xfer) != 0) {
144  return -1;
145  }
146  status = 0;
147  }
148  } while ( status != 0x02);
149 
150  return 0;
151 }

181 uint32_t w25qxx_read_id(W25QXX_DEF_PTR dev)
182 {
183  uint32_t id = 0;
184  uint8_t local_buf[4];
185  DEV_SPI_TRANSFER cmd_xfer;
186 
187  local_buf[0] = RDID;
188 
189  DEV_SPI_XFER_SET_TXBUF(&cmd_xfer, local_buf, 0, 1);
190  DEV_SPI_XFER_SET_RXBUF(&cmd_xfer, local_buf, 1, 3);
191  DEV_SPI_XFER_SET_NEXT(&cmd_xfer, NULL);
192 
193  if (spi_send_cmd(dev, &cmd_xfer) == 0) {
194  id = (local_buf[0] << 16) | (local_buf[1] << 8) | local_buf[2];
195  } else {

197  }

261 int32_t w25qxx_erase(W25QXX_DEF_PTR dev, uint32_t address, uint32_t size)
262 {
263  uint32_t last_address;
264  uint32_t count = 0;
265  uint8_t local_buf[4];
266  DEV_SPI_TRANSFER cmd_xfer;
267 
268  // start address of last sector
269  last_address = (address + size) & (~(dev->sector_sz - 1));
270 
271  // start address of first sector
272  address &= ~(dev->sector_sz - 1);
273 
274  do {
275  if (w25qxx_write_enable(dev) != 0) {
276  return -1;
277  }
278 
279  if (w25qxx_wait_ready(dev) != 0) {
280  return -1;
281  }
282 
283  local_buf[0] = SE;
284  local_buf[1] = (address >> 16) & 0xff;
285  local_buf[2] = (address >> 8) & 0xff;
286  local_buf[3] = address & 0xff;
287 
288  DEV_SPI_XFER_SET_TXBUF(&cmd_xfer, local_buf, 0, 4);
289  DEV_SPI_XFER_SET_RXBUF(&cmd_xfer, NULL, 0, 0);
290  DEV_SPI_XFER_SET_NEXT(&cmd_xfer, NULL);
291 
292  if (spi_send_cmd(dev, &cmd_xfer) != 0) {
293  return -1;
294  }
295 
296  address += dev->sector_sz;
297  count++;
298  } while(address <= last_address);
299 
300  if (w25qxx_wait_ready(dev) != 0) {
301  return -1;
302  }
303 
304  return (int32_t)count;
305 }

317 int32_t w25qxx_write(W25QXX_DEF_PTR dev, uint32_t address, uint32_t size, const void *data)
318 {
319 
320  uint8_t local_buf[4];
321  DEV_SPI_TRANSFER cmd_xfer;
322  DEV_SPI_TRANSFER data_xfer;
323 
324  uint32_t first = 0;
325  uint32_t size_orig = size;
326 
327  if (w25qxx_wait_ready(dev) != 0) {
328  return -1;
329  }
330 
331  first = FLASH_PAGE_SIZE - (address & (FLASH_PAGE_SIZE - 1));
332 
333  do {
334  // send write enable command to flash
335  if (w25qxx_write_enable(dev) != 0) {
336  return -1;
337  }
338  if (w25qxx_wait_ready(dev) != 0) {
339  return -1;
340  }
341 
342  first = first < size ? first : size;
343 
344  local_buf[0] = PP;
345  local_buf[1] = (address >> 16) & 0xff;
346  local_buf[2] = (address >> 8) & 0xff;
347  local_buf[3] = address & 0xff;
348 
349  DEV_SPI_XFER_SET_TXBUF(&data_xfer, data, 0, first);
350  DEV_SPI_XFER_SET_RXBUF(&data_xfer, NULL, 0, 0);
351  DEV_SPI_XFER_SET_NEXT(&data_xfer, NULL);
352 
353  DEV_SPI_XFER_SET_TXBUF(&cmd_xfer, local_buf, 0, 4);
354  DEV_SPI_XFER_SET_RXBUF(&cmd_xfer, NULL, 0, 0);
355  DEV_SPI_XFER_SET_NEXT(&cmd_xfer, &data_xfer);
356 
357  if (spi_send_cmd(dev, &cmd_xfer) != 0) {
358  return -1;
359  }
360 
361  size -= first;
362  address += first;
363  data += first;
364  first = FLASH_PAGE_SIZE;
365 
366  } while (size);
367 
368  if (w25qxx_wait_ready(dev) != 0) {
369  return -1;
370  }
371 
372  return (int32_t)(size_orig);
373 }
