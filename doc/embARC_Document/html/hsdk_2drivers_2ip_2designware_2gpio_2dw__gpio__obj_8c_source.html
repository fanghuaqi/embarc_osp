87 static int32_t cpu_porta_control(uint32_t ctrl_cmd, void *param)
88 {
89  uint32_t val32;
90  uint32_t i;
91  switch (ctrl_cmd) {
92  case GPIO_CMD_ENA_BIT_INT:
93  val32 = (uint32_t)param;
94  for (i = 0; i < cpu_dw_gpio_bit_isr_a.int_bit_max_cnt; i ++) {
95  if ((1<<i) & val32) {
96  int_enable(HSDC_GPIO0_ISR + i);
97  }
98  }
99  break;
100  case GPIO_CMD_DIS_BIT_INT:
101  val32 = (uint32_t)param;
102  for (i = 0; i < cpu_dw_gpio_bit_isr_a.int_bit_max_cnt; i ++) {
103  if ((1<<i) & val32) {
104  int_disable(HSDC_GPIO0_ISR + i);
105  }
106  }
107  break;
108  default:
109  break;
110  }
111  return dw_gpio_control(&cpu_port_a, ctrl_cmd, param);
112 }

129 static void cpu_porta_install(void)
130 {
131  uint32_t i;
132  DEV_GPIO_PTR port_ptr = &cpu_port_a;
133  DEV_GPIO_INFO_PTR info_ptr = &(cpu_port_a.gpio_info);
134  DW_GPIO_PORT_PTR dw_port_ptr = &(cpu_dw_gpio_port_a);
135 
136  info_ptr->gpio_ctrl = (void *)dw_port_ptr;
137  info_ptr->opn_cnt = 0;
138  info_ptr->method = 0;
139  info_ptr->direction = 0;
140  info_ptr->extra = 0;
141 
142  dw_port_ptr->no = DW_GPIO_PORT_A;
143  dw_port_ptr->regs = (DW_GPIO_REG_PTR)(HSDC_GPIO_REGBASE);
144  dw_port_ptr->valid_bit_mask = HSDK_CPU_GPIO_A_VALID_MASK;
145  dw_port_ptr->intno = DW_GPIO_INVALID_INTNO; /* pass invalid interrupt number */
146  dw_port_ptr->int_handler = cpu_porta_isr;
147 
148  for (i=0; i < cpu_dw_gpio_bit_isr_a.int_bit_max_cnt; i++) {
149  int_handler_install(HSDC_GPIO0_ISR + i, cpu_porta_isr);
150  int_disable(HSDC_GPIO0_ISR + i);
151  cpu_dw_gpio_bit_isr_a.int_bit_handler_ptr[i] = NULL;
152  }
153  dw_port_ptr->gpio_bit_isr = &cpu_dw_gpio_bit_isr_a;
154 
155  port_ptr->gpio_open = cpu_porta_open;
156  port_ptr->gpio_close = cpu_porta_close;
157  port_ptr->gpio_control = cpu_porta_control;
158  port_ptr->gpio_write = cpu_porta_write;
159  port_ptr->gpio_read = cpu_porta_read;
160 }
