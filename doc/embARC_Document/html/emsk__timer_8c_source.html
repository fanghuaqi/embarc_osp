84 void board_timer_update(uint32_t precision)
85 {
86  static uint32_t sys_hz_update = 0;
87  static uint32_t sys_ms_update = 0;
88  uint32_t hz_conv = 0;
89 
91  hz_conv = HZ_COUNT_CONV(precision, BOARD_SYS_TIMER_HZ);
92  sys_hz_update ++;
93  if (sys_hz_update >= hz_conv) {
94  sys_hz_update = 0;
95  gl_emsk_sys_hz_cnt ++;
96  }
97 
99  hz_conv = HZ_COUNT_CONV(precision, BOARD_SYS_TIMER_MS_HZ);
100  sys_ms_update ++;
101  if (sys_ms_update >= hz_conv) {
102  sys_ms_update = 0;
103  gl_emsk_ms_cnt ++;
104 
105 #ifdef MID_FATFS
106  emsk_sdcard_1ms_update();
107 #endif
108  }
109 }

128 void emsk_timer_init(void)
129 {
130  if (timer_present(BOARD_SYS_TIMER_ID)) {
131  int_disable(BOARD_SYS_TIMER_INTNO); /* disable first then enable */
132  int_handler_install(BOARD_SYS_TIMER_INTNO, emsk_timer_isr);
133  timer_start(BOARD_SYS_TIMER_ID, TIMER_CTRL_IE|TIMER_CTRL_NH, cyc_hz_count); /* start 1ms timer interrupt */
134 
135  int_enable(BOARD_SYS_TIMER_INTNO);
136  }
137 }

159 uint64_t board_get_cur_us(void)
160 {
161  uint32_t sub_us;
162  uint64_t total_us;
163  timer_current(TIMER_0, &sub_us);
164 
165  sub_us = ((uint64_t)sub_us * 1000000) / BOARD_CPU_CLOCK;
166  total_us = ((uint64_t)OSP_GET_CUR_MS()) * 1000 + (uint64_t)sub_us;
167 
168  return total_us;
169 }

185 void board_delay_ms(uint32_t ms, uint8_t os_compat)
186 {
