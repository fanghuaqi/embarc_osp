197 static void dw_gpio_toggle_dr(DW_GPIO_PORT_PTR port, uint32_t bit_mask)
198 {
199  uint32_t temp_reg1, temp_reg2;
200 
201  temp_reg1 = port->regs->SWPORTS[port->no].DR;
202  // save unmasked bits value
203  temp_reg2 = temp_reg1 & (~bit_mask);
204  // toggle masked bits value
205  temp_reg1 = (~temp_reg1) & bit_mask;
206  // combine both masked and unmasked bits
207  temp_reg1 |= temp_reg2;
208 
209  port->regs->SWPORTS[port->no].DR = temp_reg1;
210 }

238 int32_t dw_gpio_open(DEV_GPIO *gpio_obj, uint32_t dir)
239 {
240  int32_t ercd = E_OK;
241  DEV_GPIO_INFO_PTR port_info_ptr = &(gpio_obj->gpio_info);
242 
243  /* START ERROR CHECK */
244  VALID_CHK_GPIO_INFO_OBJECT(port_info_ptr);
245  /* END OF ERROR CHECK */
246 
247  DW_GPIO_PORT_PTR port = (DW_GPIO_PORT_PTR)(port_info_ptr->gpio_ctrl);
248  DW_GPIO_CHECK_EXP(port->no <= DW_GPIO_PORT_D, E_OBJ);
249 
250  port_info_ptr->opn_cnt ++;
251  if (port_info_ptr->opn_cnt > 1) { /* opened before */
252  if (dir == port_info_ptr->direction) { /* direction is the same */
253  return E_OK;
254  } else { /* open with different direction */
255  return E_OPNED;
256  }
257  }
258 
259  dw_gpio_write_dir(port, port->valid_bit_mask, dir);
260 
261  if (port->no == DW_GPIO_PORT_A) {
262  dw_gpio_int_clear(port, DW_GPIO_MASK_ALL);
263  dw_gpio_int_disable(port, DW_GPIO_MASK_ALL);
264  dw_gpio_int_unmask(port, DW_GPIO_MASK_ALL);
265  /* install gpio interrupt handler */
266  if (port->intno != DW_GPIO_INVALID_INTNO) { // Handle Bit ISR 1-1 map to vector table
267  int_handler_install(port->intno, port->int_handler);
268  int_disable(port->intno);
269  }
271  dw_gpio_set_int_cfg(port, (DEV_GPIO_INT_CFG *)(&gpio_int_cfg_default));
272  port_info_ptr->method = dw_gpio_read_mthd(port);
273  } else {
274  port_info_ptr->method = DEV_GPIO_BITS_MTHD_DEFAULT;
275  }
276 
277  dw_gpio_write_dr(port, port->valid_bit_mask, 0);
278 
279  port_info_ptr->direction = dir;
280  port_info_ptr->extra = NULL;
281  port_info_ptr->bitofs = 0;
282 
283 error_exit:
284  return ercd;
285 }

370 int32_t dw_gpio_control(DEV_GPIO *gpio_obj, uint32_t ctrl_cmd, void *param)
371 {
372  int32_t ercd = E_OK;
373  DEV_GPIO_INFO_PTR port_info_ptr = &(gpio_obj->gpio_info);
374 
375  /* START ERROR CHECK */
376  VALID_CHK_GPIO_INFO_OBJECT(port_info_ptr);
377  /* END OF ERROR CHECK */
378 
379  DW_GPIO_PORT_PTR port = (DW_GPIO_PORT_PTR)(port_info_ptr->gpio_ctrl);
380  DW_GPIO_CHECK_EXP(port->no <= DW_GPIO_PORT_D, E_OBJ);
381  DW_GPIO_CHECK_EXP(port_info_ptr->opn_cnt > 0, E_CLSED);
382 
383  uint32_t val32;
385  if (ctrl_cmd == GPIO_CMD_SET_BIT_DIR_INPUT) {
386  val32 = (uint32_t)param;
387  dw_gpio_write_dir(port, val32, DW_GPIO_INPUT_ALL);
388  port_info_ptr->direction = dw_gpio_read_dir(port);
389  } else if (ctrl_cmd == GPIO_CMD_SET_BIT_DIR_OUTPUT) {
390  val32 = (uint32_t)param;
391  dw_gpio_write_dir(port, val32, DW_GPIO_OUTPUT_ALL);
392  port_info_ptr->direction = dw_gpio_read_dir(port);
393  } else if (ctrl_cmd == GPIO_CMD_GET_BIT_DIR) {
394  DW_GPIO_CHECK_EXP((param!=NULL) && CHECK_ALIGN_4BYTES(param), E_PAR);
395  port_info_ptr->direction = dw_gpio_read_dir(port);
396  *((int32_t *)param) = port_info_ptr->direction;
397  } else if (ctrl_cmd == GPIO_CMD_TOGGLE_BITS) {
398  val32 = (uint32_t)param;
399  // Only mask the output bits
400  val32 &= port_info_ptr->direction;
401  dw_gpio_toggle_dr(port, val32);
402  } else {
403  DW_GPIO_CHECK_EXP(port->no == DW_GPIO_PORT_A, E_NOSPT);
404  /* output pin cannot be used as interrupt */
405  DEV_GPIO_INT_CFG *gpio_int_cfg;
406  DEV_GPIO_BIT_ISR *port_bit_isr;
407 
408  switch (ctrl_cmd) {
409  case GPIO_CMD_SET_BIT_INT_CFG:
410  DW_GPIO_CHECK_EXP((param!=NULL) && CHECK_ALIGN_4BYTES(param), E_PAR);
411  gpio_int_cfg = (DEV_GPIO_INT_CFG *)param;
412  dw_gpio_set_int_cfg(port, gpio_int_cfg);
413  break;
414  case GPIO_CMD_GET_BIT_INT_CFG:
415  DW_GPIO_CHECK_EXP((param!=NULL) && CHECK_ALIGN_4BYTES(param), E_PAR);
416  gpio_int_cfg = (DEV_GPIO_INT_CFG *)param;
418  dw_gpio_get_int_cfg(port, gpio_int_cfg);
419  break;
420  case GPIO_CMD_SET_BIT_ISR:
421  DW_GPIO_CHECK_EXP((param!=NULL) && CHECK_ALIGN_4BYTES(param), E_PAR);
422  port_bit_isr = (DEV_GPIO_BIT_ISR *)param;
423  if (port_bit_isr->int_bit_ofs < port->gpio_bit_isr->int_bit_max_cnt) {
424  port->gpio_bit_isr->int_bit_handler_ptr[port_bit_isr->int_bit_ofs] = port_bit_isr->int_bit_handler;
425  } else {
426  ercd = E_PAR;
427  }
428  break;
429  case GPIO_CMD_GET_BIT_ISR:
430  DW_GPIO_CHECK_EXP((param!=NULL) && CHECK_ALIGN_4BYTES(param), E_PAR);
431  port_bit_isr = (DEV_GPIO_BIT_ISR *)param;
432  if (port_bit_isr->int_bit_ofs < port->gpio_bit_isr->int_bit_max_cnt) {
433  port_bit_isr->int_bit_handler = port->gpio_bit_isr->int_bit_handler_ptr[port_bit_isr->int_bit_ofs];
434  } else {
435  ercd = E_PAR;
436  }
437  break;
438  case GPIO_CMD_ENA_BIT_INT:
439  val32 = (uint32_t)param;
440  dw_gpio_int_enable(port, val32);
441  port_info_ptr->method = dw_gpio_read_mthd(port);
442  if (port_info_ptr->method) {
443  if (port->intno != DW_GPIO_INVALID_INTNO) {
444  int_enable(port->intno);
445  }
446  }
447  break;
448  case GPIO_CMD_DIS_BIT_INT:
449  val32 = (uint32_t)param;
450  dw_gpio_int_disable(port, val32);
451  port_info_ptr->method = dw_gpio_read_mthd(port);
452  if (port_info_ptr->method == 0) {
453  if (port->intno != DW_GPIO_INVALID_INTNO) {
454  int_disable(port->intno);
455  }
456  }
457  break;
458  case GPIO_CMD_GET_BIT_MTHD:
459  DW_GPIO_CHECK_EXP((param!=NULL) && CHECK_ALIGN_4BYTES(param), E_PAR);
460  port_info_ptr->method = dw_gpio_read_mthd(port);
461  *((int32_t *)param) = port_info_ptr->method;
462  break;
463  default:
464  ercd = E_NOSPT;
465  break;
466  }
467  }
468 error_exit:
469  return ercd;
470 }

473 int32_t dw_gpio_isr_handler(DEV_GPIO *gpio_obj, void *ptr)
474 {
475  int32_t ercd = E_OK;
476  DEV_GPIO_INFO_PTR port_info_ptr = &(gpio_obj->gpio_info);
477 
478  /* START ERROR CHECK */
479  VALID_CHK_GPIO_INFO_OBJECT(port_info_ptr);
480  /* END OF ERROR CHECK */
481 
482  DW_GPIO_PORT_PTR port = (DW_GPIO_PORT_PTR)(port_info_ptr->gpio_ctrl);
483  DW_GPIO_CHECK_EXP(port->no == DW_GPIO_PORT_A, E_NOSPT);
484 
485  uint32_t i, gpio_bit_isr_state;
486  uint32_t max_int_bit_count = 0;
487 
489  gpio_bit_isr_state = dw_gpio_int_read_status(port);
490 
491  if (port->gpio_bit_isr) {
492  max_int_bit_count = (port->gpio_bit_isr->int_bit_max_cnt);
493  } else {
494  dw_gpio_int_clear(port, gpio_bit_isr_state);
495  }
496 
497  for (i=0; i<max_int_bit_count; i++) {
498  if (gpio_bit_isr_state & (1<<i)) {
499  /* this bit interrupt enabled */
500  port_info_ptr->bitofs = i;
501  if (port->gpio_bit_isr->int_bit_handler_ptr[i]) {
502  port->gpio_bit_isr->int_bit_handler_ptr[i](gpio_obj);
503  }
504  dw_gpio_int_clear(port, (1<<i));
505  }
506  }
507 
508 error_exit:
509  return ercd;
510 }
