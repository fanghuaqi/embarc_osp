45 /* data format of rw009 */
46 /*
47  *
48  * Master: REQ(16bytes) + DUMMY + DATA to transfer
49  * RW009: DUMMY + RESP(16 bytes)+ DATA to transfer
50  *
51  * DATA: len(4bytes) + type(4bytes) + buffer
52  * type: eth_data, cmd, resp, status
53  * eth_data: ethernet datapacket to IP stack
54  * cmd & resp:cmd(4bytes) + len(4bytes) + params
55  *
56  */

344 static int32_t cmd_resp_handler(DEV_WNIC *rw009_wnic, struct rw009_resp* resp)
345 {
346  int32_t ercd = E_OK;
347  DEV_WNIC_INFO_PTR rw009_info;
348  DEV_WNIC_ON_OPS_PTR rw009_on_ops;
349  RW009_DEF_PTR rw009;
350 
351  RW009_CHECK_EXP(rw009_wnic != NULL, E_OBJ);
352  rw009_info = &(rw009_wnic->wnic_info);
353  rw009_on_ops = &(rw009_wnic->wnic_on_ops);
354 
355  rw009 = (RW009_DEF_PTR)rw009_info->ctrl;
356 
357  if (resp->cmd != rw009->last_cmd) {
358  DBG("current command is %d, cmd and response do not match!\r\n", resp->cmd);
359  }
360 
361  switch (resp->cmd) {
362  case RW009_CMD_INIT:
363  RW009_CHECK_EXP((rw009_info->init_status == WNIC_DURING_INITIALIZATION), E_PAR);
364 
365  DBG("response for RW009_CMD_INIT\r\n");
366  DBG("sn:%s\r\n", resp->resp.init.sn);
367  DBG("ver:%s\r\n", resp->resp.init.version);
368  memcpy(rw009_info->mac_addr, resp->resp.init.mac, WNIC_HDR_LEN);
369  rw009_info->init_status = WNIC_INIT_SUCCESSFUL;
370  rw009->init_try = 0;
371 
372  if (rw009_on_ops->on_init_success != NULL) {
373  rw009_on_ops->on_init_success(rw009_wnic);
374  }
375  break;
376  case RW009_CMD_SCAN:
377  RW009_CHECK_EXP((rw009_info->scan_status == WNIC_DURING_SCAN), E_PAR);
378  DBG("response for rw009_cmd_scan\n");
379  if (rw009->scan_cnt < RW009_SCAN_AP_NUM) {
380  memcpy(&(rw009->scanned_ap[rw009->scan_cnt++]), &resp->resp.ap_info, sizeof(RW009_AP_INFO_DEF));
381  }
382  break;
383  case RW009_CMD_JOIN:
384  if (rw009_info->conn_status != WNIC_CONNECTED) {
385  break;
386  }
387  case RW009_CMD_EASY_JOIN:
388  RW009_CHECK_EXP((rw009_info->init_status == WNIC_INIT_SUCCESSFUL), E_PAR);
389  RW009_CHECK_EXP((rw009_info->conn_status == WNIC_DURING_CONNECTION) || \
390  (rw009_info->conn_status == WNIC_CONNECTED), E_CLS);
391  rw009_info->busy_status = WNIC_IS_FREE;
392  DBG("response for RW009_CMD_JOIN or easy_join(cmd:%d, resp:%d)\r\n", resp->cmd, resp->result);
393  if (resp->result == 0) {
394  /* connected */
395  DBG("rw009 connection success\r\n");
396  if (rw009_info->conn_status != WNIC_CONNECTED) {
397  rw009_info->conn_status = WNIC_CONNECTED;
398  if (rw009_on_ops->on_connected != NULL) {
399  rw009_on_ops->on_connected(rw009_wnic);
400  }
401  }
402  } else {
403  if (! (((resp->cmd == RW009_CMD_JOIN) && (resp->result == 12)) || \
404  (resp->cmd == RW009_CMD_EASY_JOIN)) ) {
405  break;
406  }
407  if ((rw009_info->conn_status == WNIC_CONNECTED) && \
408  (resp->cmd == RW009_CMD_EASY_JOIN)) {
409  break;
410  }
411  DBG("rw009 connection failure\r\n");
412  if (rw009_info->conn_status == WNIC_CONNECTED) {
413  rw009_info->conn_status = WNIC_NOT_CONNECTED;
414  if (rw009_on_ops->on_disconnected != NULL) {
415  rw009_on_ops->on_disconnected(rw009_wnic);
416  }
417  } else {
418  rw009_info->conn_status = WNIC_NOT_CONNECTED;
419  }
420  }
421  break;
422  case RW009_CMD_RSSI:
423  RW009_CHECK_EXP((rw009_info->init_status == WNIC_INIT_SUCCESSFUL), E_PAR);
424  DBG("response for RW009_CMD_RSSI\r\n");
425  /* \todo use to poll the status of connection ? */
426  break;
427  default:
428  DBG("unknown cmd response\r\n");
429  ercd = E_PAR;
430  break;
431  }
432 
433 error_exit:
434  return ercd;
435 }

445 static int32_t rw009_xfer(DEV_WNIC *rw009_wnic, struct rw009_spi_data_packet* out, struct rw009_spi_data_packet* in)
446 {
447  int32_t ercd = E_OK;
448  DEV_WNIC_INFO_PTR rw009_info;
449  DEV_WNIC_ON_OPS_PTR rw009_on_ops;
450  RW009_DEF_PTR rw009;
451  DEV_SPI_PTR spi;
452 
453  struct rw009_spi_request req;
454  struct rw009_spi_response resp;
455  struct dev_spi_transfer xfer;
456  uint32_t to_send = 0;
457 
458 
459  RW009_CHECK_EXP(rw009_wnic != NULL, E_OBJ);
460  rw009_info = &(rw009_wnic->wnic_info);
461  rw009_on_ops = &(rw009_wnic->wnic_on_ops);
462 
463  if (out == NULL || in == NULL) {
464  return -1;
465  }
466 
467  rw009 = (RW009_DEF_PTR)rw009_info->ctrl;
468  spi = spi_get_dev(rw009->spi_master);
469 
470  rw009_int_disable(rw009);
471 
472  while (rw009_is_busy(rw009)); /* wait if rw009 is busy */
473 
474  req.magic1 = RW009_REQ_MAGIC1;
475  req.magic2 = RW009_REQ_MAGIC2;
476  req.flag = RW009_REQ_FLAG_MRDY;
477  if (out->len) {
478  req.len = out->len + RW009_DATA_PACKET_HDR_LEN;
479  } else {
480  req.len = 0;
481  }
482 
483  spi->spi_control(SPI_CMD_MST_SET_FREQ, CONV2VOID(rw009->spi_freq));
484  spi->spi_control(SPI_CMD_SET_CLK_MODE, CONV2VOID(SPI_CLK_MODE_0));
485 
486  /* send cmd request */
487  rw009_select(rw009);
488  spi->spi_write(&req, sizeof(req));
489  rw009_deselect(rw009);
490 
491  while (rw009_is_busy(rw009));
492 
493  rw009_select(rw009);
494 
495  /* receive cmd request response */
496  spi->spi_read(&resp, sizeof(resp));
497 
498  if ((resp.magic1 != RW009_RESP_MAGIC1) || (resp.magic2 != RW009_RESP_MAGIC2)) {
499  DBG("bad resp magic, abort!\r\n");
500  return E_SYS;
501  }
502 
503  if (resp.flag & RW009_RESP_FLAG_SRDY) {
504  //DBG("slave is ready to receive\n");
505  to_send = req.len;
506  }
507 
508  if (resp.len > RW009_SPI_MAX_DATA_LEN) {
509  resp.len = 0;
510  }
511 
512  DEV_SPI_XFER_SET_TXBUF(&xfer, out, 0, to_send);
513  DEV_SPI_XFER_SET_RXBUF(&xfer, in, 0, resp.len);
514  DEV_SPI_XFER_SET_NEXT(&xfer, NULL);
515 
516  spi->spi_control(SPI_CMD_TRANSFER_POLLING, &xfer);
517 
518  rw009_deselect(rw009);
519 
520  if (resp.len > 0) {
521  if (in->type == RW009_TYPE_ETH_DATA) {
522  //DBG("eth data:%d\n", resp.len);
523  /* ethernet data */
524  if (rw009_info->init_status == WNIC_INIT_SUCCESSFUL) {
525  if (rw009_on_ops->on_rxdata_comes != NULL) {
526  rw009_on_ops->on_rxdata_comes(rw009_wnic);
527  }
528  }
529  /* eth packet it recevied */
530  } else if (in->type == RW009_TYPE_RESP) {
531  /* cmd respone received */
532  cmd_resp_handler(rw009_wnic, (struct rw009_resp *)in->buffer);
533  } else {
534  DBG("unsupported data type\r\n");
535  ercd = E_SYS;
536  }
537  }
538  rw009_int_enable(rw009);
539  //DBG("master send %d bytes, rw009 send %d bytes\r\n", to_send, resp.len);
540 error_exit:
541  return ercd;
542 }

552 static int32_t rw009_cmd(DEV_WNIC *rw009_wnic, uint32_t cmd, void *args)
553 {
554  uint32_t cmd_len;
555 #ifdef RW009_SPI_CPULOCK_ENABLE
556  uint32_t cpu_status;
557 #endif
558  RW009_CMD_DEF_PTR cmd_packet;
559  struct rw009_spi_data_packet* out;
560  RW009_DEF_PTR rw009;
561 
562  cmd_len = 0;
563  rw009 = (RW009_DEF_PTR)(rw009_wnic->wnic_info.ctrl);
564 
565  switch (cmd) {
566  case RW009_CMD_INIT:
567  cmd_len = sizeof(RW009_CMD_INIT_DEF);
568  rw009->init_timer = rw009_timer_read_ms() + RW009_INIT_TIME_MS;
569  break;
570  case RW009_CMD_SCAN:
571  cmd_len = 0;
572  break;
573  case RW009_CMD_JOIN:
574  cmd_len = sizeof(RW009_CMD_JOIN_DEF);
575  rw009->conn_timer = rw009_timer_read_ms() + RW009_CON_TIME_MS;
576  break;
577  case RW009_CMD_EASY_JOIN:
578  cmd_len = sizeof(RW009_CMD_EASY_JOIN_DEF);
579  rw009->conn_timer = rw009_timer_read_ms() + RW009_CON_TIME_MS;
580  break;
581  case RW009_CMD_SOFTAP:
582  cmd_len = sizeof(RW009_CMD_SOFTAP_DEF);
583  break;
584  case RW009_CMD_RSSI:
585  cmd_len = sizeof(RW009_CMD_RSSI_DEF);
586  break;
587  default:
588  cmd_len = 0xff;
589  break;
590  }
591 
592  if (cmd_len == 0xff) {
593  DBG("cmd type error\r\n");
594  return -1;
595  }
596 
597 #ifdef RW009_SPI_CPULOCK_ENABLE
598  cpu_status = cpu_lock_save();
599 #endif
600 
601  out = &rw009->snd_packet;
602  cmd_packet = (RW009_CMD_DEF_PTR)out->buffer;
603 
604  cmd_packet->cmd = cmd;
605  cmd_packet->len = cmd_len;
606  memcpy(&cmd_packet->params, args, cmd_len);
607  rw009->last_cmd = cmd;
608  out->type = RW009_TYPE_CMD;
609  out->len = RW009_CMD_PACKET_HDR_LEN + cmd_len;
610  rw009_xfer(rw009_wnic, out, &rw009->rcv_packet);
611  out->len = 0;
612  out->type = 0;
613 
614 #ifdef RW009_SPI_CPULOCK_ENABLE
615  cpu_unlock_restore(cpu_status);
616 #endif
617  return 0;
618 }
