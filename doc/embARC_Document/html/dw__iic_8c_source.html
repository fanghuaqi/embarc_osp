385 static uint32_t dw_iic_get_slv_state(DW_IIC_REG *iic_reg_ptr)
386 {
387  uint32_t status;
388  uint32_t slv_state = IIC_SLAVE_STATE_FREE;
389 
390  status = iic_reg_ptr->IC_RAW_INTR_STAT;
391  if (status & IC_INTR_STAT_GEN_CALL) {
392  /* General Call address is received and it is acknowledged */
393  slv_state |= IIC_SLAVE_STATE_GC_REQ;
394  }
395  if (status & IC_INTR_STAT_RX_FULL) {
396  /* master is attempting to write data to this slave */
397  slv_state |= IIC_SLAVE_STATE_WR_REQ;
398  }
399  if (status & IC_INTR_STAT_RD_REQ) {
400  /* master is attempting to read data from this slave */
401  slv_state |= IIC_SLAVE_STATE_RD_REQ;
402  }
403  if (status & IC_INTR_STAT_RX_DONE) {
404  /* master does not acknowledge a transmitted byte, and transmission is done */
405  slv_state |= IIC_SLAVE_STATE_RD_DONE;
406  status = iic_reg_ptr->IC_CLR_RX_DONE;
407  }
408  if (status & IC_INTR_STAT_START_DET) {
409  /* a START or RESTART condition has occurred */
410  slv_state |= IIC_SLAVE_STATE_START;
411  status = iic_reg_ptr->IC_CLR_START_DET; /* Clear it when read */
412  }
413  if (status & IC_INTR_STAT_STOP_DET) {
414  /* a STOP condition has occurred */
415  slv_state |= IIC_SLAVE_STATE_STOP;
416  status = iic_reg_ptr->IC_CLR_STOP_DET; /* Clear it when read */
417  }
418  if (status & (IC_INTR_STAT_TX_ABRT|IC_INTR_STAT_TX_OVER\
419  |IC_INTR_STAT_RX_OVER|IC_INTR_STAT_RX_UNDER)) {
420  /* error case */
421  slv_state |= IIC_SLAVE_STATE_ERROR;
422  status = iic_reg_ptr->IC_CLR_TX_ABRT; /* Clear it when read */
423  status = iic_reg_ptr->IC_CLR_TX_OVER;
424  status = iic_reg_ptr->IC_CLR_RX_OVER;
425  status = iic_reg_ptr->IC_CLR_RX_UNDER;
426  }
427 
428  return slv_state;
429 }

433 {

454  }
455 #endif

500 static int32_t dw_iic_mst_chkerr(DW_IIC_CTRL *iic_ctrl_ptr)
501 {
502  uint32_t status;
503  int32_t ercd = IIC_ERR_NONE;
504  DW_IIC_REG *iic_reg_ptr = (DW_IIC_REG *)(iic_ctrl_ptr->dw_iic_regs);
505 
506  status = iic_reg_ptr->IC_RAW_INTR_STAT;
507  if (status & IC_INTR_STAT_TX_ABRT) {
508  status = iic_reg_ptr->IC_TX_ABRT_SOURCE;
509  if (status & IIC_MST_ABRT_LOST_BUS) {
510  ercd = IIC_ERR_LOST_BUS;
511  } else if (status & IIC_MST_ABRT_ADDR_NOACK) {
512  ercd = IIC_ERR_ADDR_NOACK;
513  } else if (status & IIC_MST_ABRT_DATA_NOACK) {
514  ercd = IIC_ERR_DATA_NOACK;
515  } else {
516  ercd = IIC_ERR_UNDEF;
517  }
518  status = iic_reg_ptr->IC_CLR_TX_ABRT;
519  } else {
