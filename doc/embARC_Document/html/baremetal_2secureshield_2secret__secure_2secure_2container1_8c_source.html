100 // Note that some parameters of this function are pointers. Care should be taken that Container 1 has access to the memory pointed at
101 int32_t operate_secret(char* pwd, uint32_t cmd, char * data)
102 {
103  SECRET_CONTEXT *ctx;
104  uint32_t len;
105  int32_t caller_container;
106 
107  ctx = &container1_context;
108 
109  if (ctx->initialized == 0) {
110  return -1;
111  }
112 
113  // protect against brute force password guessing; for better protection, number of fails should be stored persistently.
114  if (ctx->fails++ > 3) {
115  EMBARC_PRINTF("Too many failed attempts to obtain secret.\r\n");
116  // optionally, the secret could be erased here
117  return -1;
118  }
119 
120  caller_container = secure_container_id_caller();
121 
122  if (caller_container < 0) {
123  EMBARC_PRINTF("containre_id_caller error\r\n");
124  return -1;
125  } else if (caller_container == 0) { // Check if caller is authorized to access the secret
126  if (strcmp(pwd, (const char *)ctx->pwd) != 0) {
127  return -1;
128  }
129  memset(pwd, 0, strlen(pwd));
130  } else {
131  EMBARC_PRINTF("container2 is trusted, no need to check password\r\n");
132  }
133 
134  // Caller is trusted now, perform the requested operation
135  ctx->fails = 0;
136  if (cmd == GET_SECRET) {
137  // note that data is copied with Container 1 privileges to the memory pointer to by the 'data' argument
138  // this could be a security risk, but since the caller is trusted the data pointer is trusted as well
139  memcpy(data, ctx->secret, SECRET_LEN);
140  } else if (cmd == SET_SECRET) {
141  memset(ctx->secret, 0, SECRET_LEN);
142  len = strlen(data);
143  if (len > SECRET_LEN) {
144  len = SECRET_LEN;
145  }
146  memcpy(ctx->secret, data, len);
147  memset(data, 0, len);
148  } else if (cmd == SET_PWD) {
149  memset(ctx->pwd, 0, PWD_LEN);
150  len = strlen(data);
151  if (len > PWD_LEN) {
152  len = PWD_LEN;
153  }
154  memcpy(ctx->pwd, data, len);
155  memset(data, 0, len);
156  } else {
157  return -1;
158  }
159 
160  return 0;
161 }
