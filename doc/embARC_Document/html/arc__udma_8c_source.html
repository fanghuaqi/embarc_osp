297 static dma_status_t _dmac_wait_channel(uint32_t channel)
298 {
299  uint32_t status;
300  status = _arc_aux_read(AUX_DMACSTAT1);
301  if (_arc_usually(status & DMACHANNEL(channel))) {
302  if (_arc_rarely(status & (DMACHANNEL(channel) << 16))) {
303  return DMA_ERROR;
304  } else {
305  status = _arc_aux_read(AUX_DMACSTAT0);
306  if (_arc_rarely(status & DMACHANNEL(channel))) {
307  return DMA_BUSY;
308  } else {
309  return DMA_IDLE;
310  }
311  }
312  }
313  return DMA_BUSY;
314 }

329 static uint32_t _dmac_memory_addr_gap(uint32_t dmac_mode)
330 {
331  uint32_t size = (dmac_mode & DMACTRLx_SIZE_MASK) >> DMACTRLx_SIZE_OFS;
332  uint32_t dwinc = (dmac_mode & DMACTRLx_DWINC_MASK) >> DMACTRLx_DWINC_OFS;
333 
334  switch (dwinc) {
335  case DMA_DW1INC2:
336  size <<= 1;
337  break;
338  case DMA_DW1INC4:
339  size <<= 2;
340  break;
341  case DMA_DW2INC2:
342  size &= ~0x1;
343  break;
344  case DMA_DW2INC4:
345  size &= ~0x1;
346  size <<= 1;
347  break;
348  case DMA_DW4INC4:
349  size &= ~0x3;
350  break;
351  case DMA_DW8INC8:
352  size &= ~0x7;
353  break;
354  case DMA_DW1INC1:
355  case DMA_DWINC_CLR:
356  default:
357  break;
358  }
359  return size;
360 }
361 
362 static uint32_t _dmac_aux_addr_gap(uint32_t dmac_mode)
363 {
364  uint32_t size = (dmac_mode & DMACTRLx_SIZE_MASK) >> DMACTRLx_SIZE_OFS;
365  uint32_t dwinc = (dmac_mode & DMACTRLx_DWINC_MASK) >> DMACTRLx_DWINC_OFS;
366 
367  switch (dwinc) {
368  case DMA_DW2INC2:
369  case DMA_DW2INC4:
370  size >>= 1;
371  break;
372  case DMA_DW4INC4:
373  case DMA_DW8INC8:
374  size >>= 2;
375  break;
376  case DMA_DW1INC1:
377  case DMA_DW1INC2:
378  case DMA_DW1INC4:
379  case DMA_DWINC_CLR:
380  default:
381  break;
382  }
383  return size;
384 }

388 #ifndef USE_DMA_CALC_ADDR /* converted to 1 function called _dmac_calc_addr */
389 static uint32_t _dmac_calc_dst_endaddr(uint32_t dst_addr, uint32_t dmac_mode)
390 {
391  /*
392  * Destination address is not incremented
393  */
394  if (_arc_rarely((dmac_mode & DMACTRLx_AM(2)) == 0)) {
395  return dst_addr;
396  }
397 
398  if (_arc_rarely(DMACTRLx_DTT(0x1) & dmac_mode)) { /* Destination is Auxiliary */
399  dst_addr = dst_addr + _dmac_aux_addr_gap(dmac_mode);
400  } else { /* Destination is Memory */
401  dst_addr = dst_addr + _dmac_memory_addr_gap(dmac_mode);
402  }
403  return dst_addr;
404 }

406 static uint32_t _dmac_calc_src_endaddr(uint32_t src_addr, uint32_t dmac_mode)
407 {
408  /*
409  * Source address is not incremented
410  */
411  if (_arc_rarely((dmac_mode & DMACTRLx_AM(1)) == 0)) {
412  return src_addr;
413  }
414 
415  if (_arc_rarely(DMACTRLx_DTT(0x2) & dmac_mode)) { /* Source is Auxiliary */
416  src_addr = src_addr + _dmac_aux_addr_gap(dmac_mode);
417  } else { /* Source is Memory */
418  src_addr = src_addr + _dmac_memory_addr_gap(dmac_mode);
419  }
420 
421  return src_addr;
422 }
423 #else
424 static void _dmac_calc_addr(uint32_t *src_addr, uint32_t *dst_addr, uint32_t mode)
425 {
426  if (_arc_usually(mode & DMACTRLx_AM(1))) { /* Source address incremented */
427  if (_arc_rarely(DMACTRLx_DTT(0x2) & mode)) { /* Source is Auxiliary */
428  *src_addr += _dmac_aux_addr_gap(mode);
429  } else { /* Source is Memory */
430  *src_addr += _dmac_memory_addr_gap(mode);
431  }
432  }
433  if (_arc_usually(mode & DMACTRLx_AM(2))) { /* Destination address incremented */
434  if (_arc_rarely(DMACTRLx_DTT(0x1) & mode)) { /* Destination is Auxiliary */
435  *dst_addr += _dmac_aux_addr_gap(mode);
436  } else { /* Destination is Memory */
437  *dst_addr += _dmac_memory_addr_gap(mode);
438  }
439  }
440 }

443 static void _dmac_set_desc(dma_desc_t * desc, void *source, void *dest, uint32_t size,
444  uint32_t dmac_mode)
445 {
446  uint32_t src_addr, dst_addr;
447 
448  size = DMA_CTRL_BLKSZ(size);
449  dmac_mode &= ~DMACTRLx_SIZE_MASK;
450  dmac_mode |= DMACTRLx_SIZE(size);
451 
452 #ifndef USE_DMA_CALC_ADDR
453  src_addr = _dmac_calc_src_endaddr((uint32_t)source, dmac_mode);
454  dst_addr = _dmac_calc_dst_endaddr((uint32_t)dest, dmac_mode);
455 #else /* use dma_calc_addr */
456  src_addr = (uint32_t)source;
457  dst_addr = (uint32_t)dest;
458  _dmac_calc_addr(&src_addr, &dst_addr, dmac_mode);
459 #endif
460  desc->DMACTRLx = dmac_mode;
461  desc->DMASARx = src_addr;
462  desc->DMADARx = dst_addr;
463  desc->DMALLPx = (uint32_t) NULL;
464 }

466 static void _dmac_fill_descriptor(uint32_t channel, dma_desc_t * desc)
467 {
468  if (_arc_usually(DMA_CHECK_REGISTER(channel))) {
469  uint32_t channel_shift = 3 * channel;
470  _arc_aux_write(AUX_DMACTRL0 + channel_shift, desc->DMACTRLx);
471  _arc_aux_write(AUX_DMASAR0 + channel_shift, desc->DMASARx);
472  _arc_aux_write(AUX_DMADAR0 + channel_shift, desc->DMADARx);
473 #if CORE_DMAC_INTERNAL_VERSION > 1
474  uint32_t *dma_llps = (uint32_t *) _arc_aux_read(AUX_DMACBASE);
475  dma_llps[channel] = desc->DMALLPx;
476  _MEMORY_FENCE();
477  //_arc_write_uncached_32((void *)(&dma_llps[channel]), desc->DMALLPx);
478  _DCACHE_FLUSH_MLINES((void *)(&dma_llps[channel]), sizeof(uint32_t));
479 #endif
480  } else {
481 #if CORE_DMAC_INTERNAL_VERSION > 1
482  uint32_t *dma_ptrs = (uint32_t *) _arc_aux_read(AUX_DMACBASE);
483  dma_ptrs[channel] = (uint32_t)desc;
484  _MEMORY_FENCE();
485  //_arc_write_uncached_32((void *)(&dma_ptrs[channel]), (uint32_t)desc);
486  _DCACHE_FLUSH_MLINES((void *)(&dma_ptrs[channel]), sizeof(uint32_t));
487 #else
488  dma_desc_t *dmac_desc = (dma_desc_t *) _arc_aux_read(AUX_DMACBASE);
489  dmac_desc = (dma_desc_t *) (&dmac_desc[channel]);
490  /* more efficient to let compiler do this copy */
491  *dmac_desc = *desc;
492  _MEMORY_FENCE();
493  _DCACHE_FLUSH_MLINES((void *)dmac_desc, sizeof(dma_desc_t));
494 #endif
495  }

694 static void dmac_process_desc(dma_desc_t * desc, uint32_t int_enable)
695 {
696  do {
697  if (_arc_rarely(desc->DMALLPx)) {
698  /*
699  * Force it to be auto-linked transfer when it is not
700  * DMA_MANUAL_LINKED_TRANSFER
701  */
702  if (_arc_usually(((desc->DMACTRLx & DMACTRLx_OP_MASK) >> DMACTRLx_OP_OFS) !=
703  DMA_MANUAL_LINKED_TRANSFER)) {
704  desc->DMACTRLx &= ~DMACTRLx_OP_MASK;
705  desc->DMACTRLx |= DMACTRLx_OP(DMA_AUTO_LINKED_TRANSFER);
706  }
707  } else {
708  desc->DMACTRLx &= ~DMACTRLx_OP_MASK;
709  desc->DMACTRLx |= DMACTRLx_OP(DMA_SINGLE_TRANSFER);
710  /* Only forcely change the interrupt bit of the last dma descriptor */
711  desc->DMACTRLx &= ~DMACTRLx_INT_MASK;
712  desc->DMACTRLx |= DMACTRLx_INT(int_enable);
713  }
714  /*
715  * TODO: Make sure all linked DMA channel descriptors
716  * stored into memory, not just in data cache
717  */
718  _MEMORY_FENCE();
719  _DCACHE_FLUSH_MLINES((void *)desc, sizeof(dma_desc_t));
720  desc = (dma_desc_t *) desc->DMALLPx;
721  } while (desc != 0);
722 }

787  _dmac_clear_error(channel);

789  _dmac_reset_channel(channel);
790 #endif

792  _dmac_enable_channel(channel);

795  * only trigger when software
796  */
797  if (_arc_usually(dma_chn->source == DMA_REQ_SOFT)) {
